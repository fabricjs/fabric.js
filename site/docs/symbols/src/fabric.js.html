 16  */

 18 
 19 if (typeof console !== 'undefined') {
 20   if (typeof console.log !== 'undefined' && console.log.apply) {
 21     fabric.log = function() {

 23     };
 24   }
 25   if (typeof console.warn !== 'undefined' && console.warn.apply) {
 26     fabric.warn = function() {

 28     };
 29   }
