 37   // Copied from Inkscape svgtopdf, thanks!
 38   function arcToSegments(x, y, rx, ry, large, sweep, rotateX, ox, oy) {
 39     argsString = _join.call(arguments);
 40     if (arcToSegmentsCache[argsString]) {
 41       return arcToSegmentsCache[argsString];
 42     }
 43     
 44     var th = rotateX * (Math.PI/180);
 45     var sin_th = Math.sin(th);
 46     var cos_th = Math.cos(th);
 47     rx = Math.abs(rx);
 48     ry = Math.abs(ry);
 49     var px = cos_th * (ox - x) * 0.5 + sin_th * (oy - y) * 0.5;
 50     var py = cos_th * (oy - y) * 0.5 - sin_th * (ox - x) * 0.5;
 51     var pl = (px*px) / (rx*rx) + (py*py) / (ry*ry);
 52     if (pl > 1) {
 53       pl = Math.sqrt(pl);
 54       rx *= pl;
 55       ry *= pl;
 56     }
 57 
 58     var a00 = cos_th / rx;
 59     var a01 = sin_th / rx;
 60     var a10 = (-sin_th) / ry;
 61     var a11 = (cos_th) / ry;
 62     var x0 = a00 * ox + a01 * oy;
 63     var y0 = a10 * ox + a11 * oy;
 64     var x1 = a00 * x + a01 * y;
 65     var y1 = a10 * x + a11 * y;
 66 
 67     var d = (x1-x0) * (x1-x0) + (y1-y0) * (y1-y0);
 68     var sfactor_sq = 1 / d - 0.25;
 69     if (sfactor_sq < 0) sfactor_sq = 0;
 70     var sfactor = Math.sqrt(sfactor_sq);
 71     if (sweep == large) sfactor = -sfactor;
 72     var xc = 0.5 * (x0 + x1) - sfactor * (y1-y0);
 73     var yc = 0.5 * (y0 + y1) + sfactor * (x1-x0);
 74 
 75     var th0 = Math.atan2(y0-yc, x0-xc);
 76     var th1 = Math.atan2(y1-yc, x1-xc);
 77 
 78     var th_arc = th1-th0;
 79     if (th_arc < 0 && sweep == 1){
 80       th_arc += 2*Math.PI;
 81     } else if (th_arc > 0 && sweep == 0) {
 82       th_arc -= 2 * Math.PI;
 83     }
 84 
 85     var segments = Math.ceil(Math.abs(th_arc / (Math.PI * 0.5 + 0.001)));
 86     var result = [];
 87     for (var i=0; i<segments; i++) {
 88       var th2 = th0 + i * th_arc / segments;
 89       var th3 = th0 + (i+1) * th_arc / segments;
 90       result[i] = [xc, yc, th2, th3, rx, ry, sin_th, cos_th];
 91     }
 92 
 93     return (arcToSegmentsCache[argsString] = result);
 94   }

 96   function segmentToBezier(cx, cy, th0, th1, rx, ry, sin_th, cos_th) {
 97     argsString = _join.call(arguments);
 98     if (segmentToBezierCache[argsString]) {
 99       return segmentToBezierCache[argsString];
100     }
101     
102     var a00 = cos_th * rx;
103     var a01 = -sin_th * ry;
104     var a10 = sin_th * rx;
105     var a11 = cos_th * ry;
106 
107     var th_half = 0.5 * (th1 - th0);
108     var t = (8/3) * Math.sin(th_half * 0.5) * Math.sin(th_half * 0.5) / Math.sin(th_half);
109     var x1 = cx + Math.cos(th0) - t * Math.sin(th0);
110     var y1 = cy + Math.sin(th0) + t * Math.cos(th0);
111     var x3 = cx + Math.cos(th1);
112     var y3 = cy + Math.sin(th1);
113     var x2 = x3 + t * Math.sin(th1);
114     var y2 = y3 - t * Math.cos(th1);
115     
116     return (segmentToBezierCache[argsString] = [
117       a00 * x1 + a01 * y1,      a10 * x1 + a11 * y1,
118       a00 * x2 + a01 * y2,      a10 * x2 + a11 * y2,
119       a00 * x3 + a01 * y3,      a10 * x3 + a11 * y3
120     ]);

230     _render: function(ctx) {
231       var current, // current instruction 
232           x = 0, // current x 
233           y = 0, // current y
234           controlX = 0, // current control point x
235           controlY = 0, // current control point y
236           tempX, 
237           tempY,
238           l = -(this.width / 2),
239           t = -(this.height / 2);
240           
241       for (var i = 0, len = this.path.length; i < len; ++i) {
242         
243         current = this.path[i];
244         
245         switch (current[0]) { // first letter
246           
247           case 'l': // lineto, relative
248             x += current[1];
249             y += current[2];
250             ctx.lineTo(x + l, y + t);
251             break;
252             
253           case 'L': // lineto, absolute
254             x = current[1];
255             y = current[2];
256             ctx.lineTo(x + l, y + t);
257             break;
258             
259           case 'h': // horizontal lineto, relative
260             x += current[1];
261             ctx.lineTo(x + l, y + t);
262             break;
263             
264           case 'H': // horizontal lineto, absolute
265             x = current[1];
266             ctx.lineTo(x + l, y + t);
267             break;
268             
269           case 'v': // vertical lineto, relative
270             y += current[1];
271             ctx.lineTo(x + l, y + t);
272             break;
273             
274           case 'V': // verical lineto, absolute
275             y = current[1];
276             ctx.lineTo(x + l, y + t);
277             break;
278             
279           case 'm': // moveTo, relative
280             x += current[1];
281             y += current[2];
282             ctx.moveTo(x + l, y + t);
283             break;
284           
285           case 'M': // moveTo, absolute
286             x = current[1];
287             y = current[2];
288             ctx.moveTo(x + l, y + t);
289             break;
290             
291           case 'c': // bezierCurveTo, relative
292             tempX = x + current[5];
293             tempY = y + current[6];
294             controlX = x + current[3];
295             controlY = y + current[4];
296             ctx.bezierCurveTo(
297               x + current[1] + l, // x1
298               y + current[2] + t, // y1
299               controlX + l, // x2
300               controlY + t, // y2
301               tempX + l,
302               tempY + t
303             );
304             x = tempX;
305             y = tempY;
306             break;
307             
308           case 'C': // bezierCurveTo, absolute
309             x = current[5];
310             y = current[6];
311             controlX = current[3];
312             controlY = current[4];
313             ctx.bezierCurveTo(
314               current[1] + l, 
315               current[2] + t, 
316               controlX + l, 
317               controlY + t, 
318               x + l, 
319               y + t
320             );
321             break;
322           
323           case 's': // shorthand cubic bezierCurveTo, relative
324             // transform to absolute x,y
325             tempX = x + current[3];
326             tempY = y + current[4];
327             // calculate reflection of previous control points            
328             controlX = 2 * x - controlX;
329             controlY = 2 * y - controlY;
330             ctx.bezierCurveTo(
331               controlX + l,
332               controlY + t,
333               x + current[1] + l,
334               y + current[2] + t,
335               tempX + l,
336               tempY + t
337             );
338             x = tempX;
339             y = tempY;
340             break;
341             
342           case 'S': // shorthand cubic bezierCurveTo, absolute
343             tempX = current[3];
344             tempY = current[4];
345             // calculate reflection of previous control points            
346             controlX = 2*x - controlX;
347             controlY = 2*y - controlY;
348             ctx.bezierCurveTo(
349               controlX + l,
350               controlY + t,
351               current[1] + l,
352               current[2] + t,
353               tempX + l,
354               tempY + t
355             );
356             x = tempX;
357             y = tempY;
358             break;
359             
360           case 'q': // quadraticCurveTo, relative
361             x += current[3];
362             y += current[4];
363             ctx.quadraticCurveTo(
364               current[1] + l, 
365               current[2] + t, 
366               x + l, 
367               y + t
368             );
369             break;
370             
371           case 'Q': // quadraticCurveTo, absolute
372             x = current[3];
373             y = current[4];
374             controlX = current[1];
375             controlY = current[2];
376             ctx.quadraticCurveTo(
377               controlX + l,
378               controlY + t,
379               x + l,
380               y + t
381             );
382             break;
383           
384           case 'T':
385             tempX = x;
386             tempY = y;
387             x = current[1];
388             y = current[2];
389             // calculate reflection of previous control points
390             controlX = -controlX + 2 * tempX;
391             controlY = -controlY + 2 * tempY;
392             ctx.quadraticCurveTo(
393               controlX + l,
394               controlY + t,
395               x + l, 
396               y + t
397             );
398             break;
399             
400           case 'a':
401             // TODO: optimize this
402             drawArc(ctx, x + l, y + t, [ 
403               current[1], 
404               current[2], 
405               current[3], 
406               current[4], 
407               current[5], 
408               current[6] + x + l,
409               current[7] + y + t
410             ]);
411             x += current[6];
412             y += current[7];

423               current[6] + l,
424               current[7] + t
425             ]);
426             x = current[6];
427             y = current[7];
428             break;
429           
430           case 'z':
431           case 'Z':
432             ctx.closePath();
433             break;
434         }

460       }
461       
462       if (this.stroke) {

464       }

560      * @private
561      * @method _parsePath

563     _parsePath: function() {
564       var result = [ ],
565           currentPath, 
566           chunks,
567           parsed;
568 
569       for (var i = 0, j, chunksParsed, len = this.path.length; i < len; i++) {
570         currentPath = this.path[i];
571         chunks = currentPath.slice(1).trim().replace(/(\d)-/g, '$1###-').split(/\s|,|###/);
572         chunksParsed = [ currentPath.charAt(0) ];
573 
574         for (var j = 0, jlen = chunks.length; j < jlen; j++) {
575           parsed = parseFloat(chunks[j]);
576           if (!isNaN(parsed)) {
577             chunksParsed.push(parsed);
578           }
579         }
580 
581         var command = chunksParsed[0].toLowerCase(),
582             commandLength = commandLengths[command];
583             
584         if (chunksParsed.length - 1 > commandLength) {
585           for (var k = 1, klen = chunksParsed.length; k < klen; k += commandLength) {
586             result.push([command].concat(chunksParsed.slice(k, k + commandLength)));
587           }
588         }
589         else {
590           result.push(chunksParsed);
591         }
592       }
593       
594       return result;

597     /**
598      * @method _parseDimensions
599      */
600     _parseDimensions: function() {
601       var aX = [], 
602           aY = [], 
603           previousX, 
604           previousY, 
605           isLowerCase = false, 
606           x, 
607           y;
608       
609       this.path.forEach(function(item, i) {
610         if (item[0] !== 'H') {
611           previousX = (i === 0) ? getX(item) : getX(this.path[i-1]);
612         }
613         if (item[0] !== 'V') {
614           previousY = (i === 0) ? getY(item) : getY(this.path[i-1]);
615         }
616         
617         // lowercased letter denotes relative position; 
618         // transform to absolute
619         if (item[0] === item[0].toLowerCase()) {
620           isLowerCase = true;
621         }
622         
623         // last 2 items in an array of coordinates are the actualy x/y (except H/V);
624         // collect them
625         
626         // TODO (kangax): support relative h/v commands
627             
628         x = isLowerCase
629           ? previousX + getX(item)
630           : item[0] === 'V' 
631             ? previousX 
632             : getX(item);
633             
634         y = isLowerCase
635           ? previousY + getY(item)
636           : item[0] === 'H' 
637             ? previousY 
638             : getY(item);
639         
640         var val = parseInt(x, 10);
641         if (!isNaN(val)) aX.push(val);
642         
643         val = parseInt(y, 10);
644         if (!isNaN(val)) aY.push(val);
645         
646       }, this);
647       
648       var minX = min(aX), 
649           minY = min(aY), 
650           deltaX = 0,
651           deltaY = 0;
652       
653       var o = {
654         top: minY - deltaY,
655         left: minX - deltaX,
656         bottom: max(aY) - deltaY,
657         right: max(aX) - deltaX
658       };
659       
660       o.width = o.right - o.left;
661       o.height = o.bottom - o.top;
662       
663       return o;
664     }
