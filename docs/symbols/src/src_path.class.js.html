 36   function arcToSegments(x, y, rx, ry, large, sweep, rotateX, ox, oy) {
 37     argsString = _join.call(arguments);
 38     if (arcToSegmentsCache[argsString]) {
 39       return arcToSegmentsCache[argsString];
 40     }
 41 
 42     var th = rotateX * (Math.PI/180);
 43     var sin_th = Math.sin(th);
 44     var cos_th = Math.cos(th);
 45     rx = Math.abs(rx);
 46     ry = Math.abs(ry);
 47     var px = cos_th * (ox - x) * 0.5 + sin_th * (oy - y) * 0.5;
 48     var py = cos_th * (oy - y) * 0.5 - sin_th * (ox - x) * 0.5;
 49     var pl = (px*px) / (rx*rx) + (py*py) / (ry*ry);
 50     if (pl > 1) {
 51       pl = Math.sqrt(pl);
 52       rx *= pl;
 53       ry *= pl;
 54     }
 55 
 56     var a00 = cos_th / rx;
 57     var a01 = sin_th / rx;
 58     var a10 = (-sin_th) / ry;
 59     var a11 = (cos_th) / ry;
 60     var x0 = a00 * ox + a01 * oy;
 61     var y0 = a10 * ox + a11 * oy;
 62     var x1 = a00 * x + a01 * y;
 63     var y1 = a10 * x + a11 * y;
 64 
 65     var d = (x1-x0) * (x1-x0) + (y1-y0) * (y1-y0);
 66     var sfactor_sq = 1 / d - 0.25;
 67     if (sfactor_sq < 0) sfactor_sq = 0;
 68     var sfactor = Math.sqrt(sfactor_sq);
 69     if (sweep == large) sfactor = -sfactor;
 70     var xc = 0.5 * (x0 + x1) - sfactor * (y1-y0);
 71     var yc = 0.5 * (y0 + y1) + sfactor * (x1-x0);
 72 
 73     var th0 = Math.atan2(y0-yc, x0-xc);
 74     var th1 = Math.atan2(y1-yc, x1-xc);
 75 
 76     var th_arc = th1-th0;
 77     if (th_arc < 0 && sweep == 1){
 78       th_arc += 2*Math.PI;
 79     } else if (th_arc > 0 && sweep == 0) {
 80       th_arc -= 2 * Math.PI;
 81     }
 82 
 83     var segments = Math.ceil(Math.abs(th_arc / (Math.PI * 0.5 + 0.001)));
 84     var result = [];
 85     for (var i=0; i<segments; i++) {
 86       var th2 = th0 + i * th_arc / segments;
 87       var th3 = th0 + (i+1) * th_arc / segments;
 88       result[i] = [xc, yc, th2, th3, rx, ry, sin_th, cos_th];
 89     }
 90 
 91     return (arcToSegmentsCache[argsString] = result);
 92   }

 94   function segmentToBezier(cx, cy, th0, th1, rx, ry, sin_th, cos_th) {
 95     argsString = _join.call(arguments);
 96     if (segmentToBezierCache[argsString]) {
 97       return segmentToBezierCache[argsString];
 98     }
 99 
100     var a00 = cos_th * rx;
101     var a01 = -sin_th * ry;
102     var a10 = sin_th * rx;
103     var a11 = cos_th * ry;
104 
105     var th_half = 0.5 * (th1 - th0);
106     var t = (8/3) * Math.sin(th_half * 0.5) * Math.sin(th_half * 0.5) / Math.sin(th_half);
107     var x1 = cx + Math.cos(th0) - t * Math.sin(th0);
108     var y1 = cy + Math.sin(th0) + t * Math.cos(th0);
109     var x3 = cx + Math.cos(th1);
110     var y3 = cy + Math.sin(th1);
111     var x2 = x3 + t * Math.sin(th1);
112     var y2 = y3 - t * Math.cos(th1);
113 
114     return (segmentToBezierCache[argsString] = [
115       a00 * x1 + a01 * y1,      a10 * x1 + a11 * y1,
116       a00 * x2 + a01 * y2,      a10 * x2 + a11 * y2,
117       a00 * x3 + a01 * y3,      a10 * x3 + a11 * y3

228     _render: function(ctx) {
229       var current, // current instruction
230           x = 0, // current x
231           y = 0, // current y
232           controlX = 0, // current control point x
233           controlY = 0, // current control point y
234           tempX,
235           tempY,
236           l = -(this.width / 2),
237           t = -(this.height / 2);
238 
239       for (var i = 0, len = this.path.length; i < len; ++i) {
240 
241         current = this.path[i];
242 
243         switch (current[0]) { // first letter
244 
245           case 'l': // lineto, relative
246             x += current[1];
247             y += current[2];
248             ctx.lineTo(x + l, y + t);
249             break;
250 
251           case 'L': // lineto, absolute
252             x = current[1];
253             y = current[2];
254             ctx.lineTo(x + l, y + t);
255             break;
256 
257           case 'h': // horizontal lineto, relative
258             x += current[1];
259             ctx.lineTo(x + l, y + t);
260             break;
261 
262           case 'H': // horizontal lineto, absolute
263             x = current[1];
264             ctx.lineTo(x + l, y + t);
265             break;
266 
267           case 'v': // vertical lineto, relative
268             y += current[1];
269             ctx.lineTo(x + l, y + t);
270             break;
271 
272           case 'V': // verical lineto, absolute
273             y = current[1];
274             ctx.lineTo(x + l, y + t);
275             break;
276 
277           case 'm': // moveTo, relative
278             x += current[1];
279             y += current[2];
280             ctx.moveTo(x + l, y + t);
281             break;
282 
283           case 'M': // moveTo, absolute
284             x = current[1];
285             y = current[2];
286             ctx.moveTo(x + l, y + t);
287             break;
288 
289           case 'c': // bezierCurveTo, relative
290             tempX = x + current[5];
291             tempY = y + current[6];
292             controlX = x + current[3];
293             controlY = y + current[4];
294             ctx.bezierCurveTo(
295               x + current[1] + l, // x1
296               y + current[2] + t, // y1
297               controlX + l, // x2
298               controlY + t, // y2
299               tempX + l,
300               tempY + t
301             );
302             x = tempX;
303             y = tempY;
304             break;
305 
306           case 'C': // bezierCurveTo, absolute
307             x = current[5];
308             y = current[6];
309             controlX = current[3];
310             controlY = current[4];
311             ctx.bezierCurveTo(
312               current[1] + l,
313               current[2] + t,
314               controlX + l,
315               controlY + t,
316               x + l,
317               y + t
318             );
319             break;
320 
321           case 's': // shorthand cubic bezierCurveTo, relative
322             // transform to absolute x,y
323             tempX = x + current[3];
324             tempY = y + current[4];
325             // calculate reflection of previous control points
326             controlX = 2 * x - controlX;
327             controlY = 2 * y - controlY;
328             ctx.bezierCurveTo(
329               controlX + l,
330               controlY + t,
331               x + current[1] + l,
332               y + current[2] + t,
333               tempX + l,
334               tempY + t
335             );
336             // set control point to 2nd one of this command
337             // "... the first control point is assumed to be the reflection of the second control point on the previous command relative to the current point."
338             controlX = x + current[1];
339             controlY = y + current[2];
340 
341             x = tempX;
342             y = tempY;
343             break;
344 
345           case 'S': // shorthand cubic bezierCurveTo, absolute
346             tempX = current[3];
347             tempY = current[4];
348             // calculate reflection of previous control points
349             controlX = 2*x - controlX;
350             controlY = 2*y - controlY;
351             ctx.bezierCurveTo(
352               controlX + l,
353               controlY + t,
354               current[1] + l,
355               current[2] + t,
356               tempX + l,
357               tempY + t
358             );
359             x = tempX;
360             y = tempY;
361 
362             // set control point to 2nd one of this command
363             // "... the first control point is assumed to be the reflection of the second control point on the previous command relative to the current point."
364             controlX = current[1];
365             controlY = current[2];
366 
367             break;
368 
369           case 'q': // quadraticCurveTo, relative
370             x += current[3];
371             y += current[4];

373               current[1] + l,
374               current[2] + t,

376               y + t

378             break;
379 
380           case 'Q': // quadraticCurveTo, absolute
381             x = current[3];
382             y = current[4];
383             controlX = current[1];
384             controlY = current[2];
385             ctx.quadraticCurveTo(
386               controlX + l,
387               controlY + t,
388               x + l,
389               y + t

393           case 'T':
394             tempX = x;
395             tempY = y;
396             x = current[1];
397             y = current[2];
398             // calculate reflection of previous control points
399             controlX = -controlX + 2 * tempX;
400             controlY = -controlY + 2 * tempY;

402               controlX + l,
403               controlY + t,
404               x + l,
405               y + t

407             break;

409           case 'a':

436             y = current[7];

606     /**
607      * @method _parseDimensions
608      */
609     _parseDimensions: function() {
610       var aX = [],
611           aY = [],
612           previousX,
613           previousY,
614           isLowerCase = false,
615           x,
616           y;
617 
618       this.path.forEach(function(item, i) {
619         if (item[0] !== 'H') {
620           previousX = (i === 0) ? getX(item) : getX(this.path[i-1]);
621         }
622         if (item[0] !== 'V') {
623           previousY = (i === 0) ? getY(item) : getY(this.path[i-1]);
624         }
625 
626         // lowercased letter denotes relative position;
627         // transform to absolute
628         if (item[0] === item[0].toLowerCase()) {
629           isLowerCase = true;
630         }
631 
632         // last 2 items in an array of coordinates are the actualy x/y (except H/V);
633         // collect them
634 
635         // TODO (kangax): support relative h/v commands
636 
637         x = isLowerCase
638           ? previousX + getX(item)
639           : item[0] === 'V'
640             ? previousX
641             : getX(item);
642 
643         y = isLowerCase
644           ? previousY + getY(item)
645           : item[0] === 'H'
646             ? previousY
647             : getY(item);
648 
649         var val = parseInt(x, 10);
650         if (!isNaN(val)) aX.push(val);
651 
652         val = parseInt(y, 10);
653         if (!isNaN(val)) aY.push(val);
654 
655       }, this);
656 
657       var minX = min(aX),
658           minY = min(aY),
659           deltaX = 0,
660           deltaY = 0;
661 
662       var o = {
663         top: minY - deltaY,
664         left: minX - deltaX,
665         bottom: max(aY) - deltaY,
666         right: max(aX) - deltaX
667       };
668 
669       o.width = o.right - o.left;
670       o.height = o.bottom - o.top;
671 
672       return o;
