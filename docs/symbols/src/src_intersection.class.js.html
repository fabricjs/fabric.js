 58   fabric.Intersection.intersectLineLine = function (a1, a2, b1, b2) {
 59     var result,
 60         ua_t = (b2.x - b1.x) * (a1.y - b1.y) - (b2.y - b1.y) * (a1.x - b1.x),
 61         ub_t = (a2.x - a1.x) * (a1.y - b1.y) - (a2.y - a1.y) * (a1.x - b1.x),
 62         u_b = (b2.y - b1.y) * (a2.x - a1.x) - (b2.x - b1.x) * (a2.y - a1.y);
 63     if (u_b != 0) {
 64       var ua = ua_t / u_b,
 65           ub = ub_t / u_b;
 66       if (0 <= ua && ua <= 1 && 0 <= ub && ub <= 1) {
 67         result = new Intersection("Intersection");
 68         result.points.push(new fabric.Point(a1.x + ua * (a2.x - a1.x), a1.y + ua * (a2.y - a1.y)));
 69       }
 70       else {
 71         result = new Intersection("No Intersection");
 72       }
 73     }
 74     else {
 75       if (ua_t == 0 || ub_t == 0) {
 76         result = new Intersection("Coincident");
 77       }
 78       else {
 79         result = new Intersection("Parallel");
 80       }
 81     }
 82     return result;
