352     /**
353      * Adds objects to canvas, then renders canvas;
354      * Objects should be instances of (or inherit from) fabric.Object
355      * @method add
356      * @return {fabric.Canvas} thisArg
357      * @chainable
358      */
359     add: function () {
360       this._objects.push.apply(this._objects, arguments);
361       for (var i = arguments.length; i--; ) {
362         this.stateful && arguments[i].setupState();
363         arguments[i].setCoords();
364         this.fire('object:added', { target: arguments[i] });
365         arguments[i].fire('added');
366       }
367       this.renderOnAddition && this.renderAll();
368       return this;
369     },

371     /**
372      * Inserts an object to canvas at specified index and renders canvas.
373      * An object should be an instance of (or inherit from) fabric.Object
374      * @method insertAt
375      * @param object {Object} Object to insert
376      * @param index {Number} index to insert object at
377      * @param nonSplicing {Boolean} when `true`, no splicing (shifting) of objects occurs
378      * @return {fabric.Canvas} instance
379      */
380     insertAt: function (object, index, nonSplicing) {
381       if (nonSplicing) {
382         this._objects[index] = object;
383       }
384       else {
385         this._objects.splice(index, 0, object);
386       }
387       this.stateful && object.setupState();
388       object.setCoords();
389       this.renderAll();
390       this.fire('object:added', { target: object });
391       object.fire('added');
392       return this;

441     /**
442      * Renders both the top canvas and the secondary container canvas.
443      * @method renderAll
444      * @param allOnTop {Boolean} optional Whether we want to force all images to be rendered on the top canvas
445      * @return {fabric.Canvas} instance
446      * @chainable
447      */
448     renderAll: function (allOnTop) {
449 
450       var canvasToDrawOn = this[(allOnTop === true && this.interactive) ? 'contextTop' : 'contextContainer'];
451 
452       if (this.contextTop) {
453         this.clearContext(this.contextTop);
454       }
455 
456       if (allOnTop === false || (typeof allOnTop === 'undefined')) {
457         this.clearContext(canvasToDrawOn);
458       }
459 
460       var length = this._objects.length,
461           activeGroup = this.getActiveGroup(),
462           startTime = new Date();
463 
464       if (this.clipTo) {
465         canvasToDrawOn.save();
466         canvasToDrawOn.beginPath();
467         this.clipTo(canvasToDrawOn);
468         canvasToDrawOn.clip();
469       }
470 
471       canvasToDrawOn.fillStyle = this.backgroundColor;
472       canvasToDrawOn.fillRect(0, 0, this.width, this.height);
473 
474       if (typeof this.backgroundImage == 'object') {
475         canvasToDrawOn.save();
476         canvasToDrawOn.globalAlpha = this.backgroundImageOpacity;
477 
478         if (this.backgroundImageStretch) {
479             canvasToDrawOn.drawImage(this.backgroundImage, 0, 0, this.width, this.height);
480         }
481         else {
482             canvasToDrawOn.drawImage(this.backgroundImage, 0, 0);
483         }
484         canvasToDrawOn.restore();
485       }
486 
487       if (length) {
488         for (var i = 0; i < length; ++i) {
489           if (!activeGroup ||
490               (activeGroup && this._objects[i] && !activeGroup.contains(this._objects[i]))) {
491             this._draw(canvasToDrawOn, this._objects[i]);
492           }
493         }
494       }
495 
496       if (this.clipTo) {
497         canvasToDrawOn.restore();
498       }
499 
500       // delegate rendering to group selection (if one exists)
501       if (activeGroup) {
502         this._draw(this.contextTop, activeGroup);
503       }
504 
505       if (this.overlayImage) {
506         (this.contextTop || this.contextContainer).drawImage(this.overlayImage, 0, 0);
507       }
508 
509       if (this.onFpsUpdate) {
510         var elapsedTime = new Date() - startTime;
511         this.onFpsUpdate(~~(1000 / elapsedTime));
512       }
513 
514       this.fire('after:render');
515 
516       return this;
517     },
518 
519     /**
520      * Method to render only the top canvas.
521      * Also used to render the group selection box.
522      * @method renderTop
523      * @return {fabric.Canvas} thisArg
524      * @chainable
525      */
526     renderTop: function () {
527       this.clearContext(this.contextTop || this.contextContainer);
528 
529       if (this.overlayImage) {
530         (this.contextTop || this.contextContainer).drawImage(this.overlayImage, 0, 0);
531       }
532 
533       // we render the top context - last object
534       if (this.selection && this._groupSelector) {
535         this._drawSelection();
536       }
537 
538       // delegate rendering to group selection if one exists
539       // used for drawing selection borders/corners
540       var activeGroup = this.getActiveGroup();
541       if (activeGroup) {
542         activeGroup.render(this.contextTop);
543       }
544 
545       this.fire('after:render');
546 
547       return this;
548     },

563     /**
564      * Exports canvas element to a dataurl image (allowing to change image size via multiplier).
565      * @method toDataURLWithMultiplier
566      * @param {String} format (png|jpeg)
567      * @param {Number} multiplier
568      * @return {String}
569      */
570     toDataURLWithMultiplier: function (format, multiplier) {
571 
572       var origWidth = this.getWidth(),
573           origHeight = this.getHeight(),
574           scaledWidth = origWidth * multiplier,
575           scaledHeight = origHeight * multiplier,
576           activeObject = this.getActiveObject(),
577           activeGroup = this.getActiveGroup();
578 
579       this.setWidth(scaledWidth).setHeight(scaledHeight);
580       this.contextTop.scale(multiplier, multiplier);
581 
582       if (activeGroup) {
583         // not removing group due to complications with restoring it with correct state afterwords
584         this._tempRemoveBordersCornersFromGroup(activeGroup);
585       }
586       else if (activeObject) {
587         this.deactivateAll();
588       }
589 
590       // restoring width, height for `renderAll` to draw
591       // background properly (while context is scaled)
592       this.width = origWidth;
593       this.height = origHeight;
594 
595       this.renderAll(true);
596 
597       var dataURL = this.toDataURL(format);
598 
599       this.contextTop.scale(1 / multiplier,  1 / multiplier);
600       this.setWidth(origWidth).setHeight(origHeight);
601 
602       if (activeGroup) {
603         this._restoreBordersCornersOnGroup(activeGroup);
604       }
605       else if (activeObject) {
606         this.setActiveObject(activeObject);
607       }
608 
609       this.renderAll();
610 
611       return dataURL;
612     },

712     /**
713      * @private
714      * @method _toObjectMethod
715      */
716     _toObjectMethod: function (methodName) {
717       var data = {
718         objects: this._objects.map(function (instance) {
719           // TODO (kangax): figure out how to clean this up
720           if (!this.includeDefaultValues) {
721             var originalValue = instance.includeDefaultValues;
722             instance.includeDefaultValues = false;
723           }
724           var object = instance[methodName]();
725           if (!this.includeDefaultValues) {
726             instance.includeDefaultValues = originalValue;
727           }
728           return object;
729         }, this),
730         background: this.backgroundColor
731       };
732       if (this.backgroundImage) {
733         data.backgroundImage = this.backgroundImage.src;
734         data.backgroundImageOpacity = this.backgroundImageOpacity;
735         data.backgroundImageStretch = this.backgroundImageStretch;
736       }
737       return data;
738     },

791     /**

793      * @method remove
