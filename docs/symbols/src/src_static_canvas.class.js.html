417     /**
418      * Renders both the top canvas and the secondary container canvas.
419      * @method renderAll
420      * @param allOnTop {Boolean} optional Whether we want to force all images to be rendered on the top canvas
421      * @return {fabric.Canvas} instance
422      * @chainable
423      */ 
424     renderAll: function (allOnTop) {
425       
426       var canvasToDrawOn = this[(allOnTop && this.interactive) ? 'contextTop' : 'contextContainer'];
427 
428       if (this.contextTop) {
429         this.clearContext(this.contextTop);
430       }
431 
432       if (!allOnTop) {
433         this.clearContext(canvasToDrawOn);
434       }
435       
436       var length = this._objects.length,
437           activeGroup = this.getActiveGroup(),
438           startTime = new Date();
439       
440       if (this.clipTo) {
441         canvasToDrawOn.save();
442         canvasToDrawOn.beginPath();
443         this.clipTo(canvasToDrawOn);
444         canvasToDrawOn.clip();
445       }
446       
447       canvasToDrawOn.fillStyle = this.backgroundColor;
448       canvasToDrawOn.fillRect(0, 0, this.width, this.height);
449       
450       if (typeof this.backgroundImage == 'object') {
451         canvasToDrawOn.drawImage(this.backgroundImage, 0, 0, this.width, this.height);
452       }
453       
454       if (length) {
455         for (var i = 0; i < length; ++i) {
456           if (!activeGroup ||
457               (activeGroup && this._objects[i] && !activeGroup.contains(this._objects[i]))) {
458             this._draw(canvasToDrawOn, this._objects[i]);
459           }
460         }
461       }
462       
463       if (this.clipTo) {
464         canvasToDrawOn.restore();
465       }
466       
467       // delegate rendering to group selection (if one exists)
468       if (activeGroup) {
469         this._draw(this.contextTop, activeGroup);
470       }
471       
472       if (this.overlayImage) {
473         this.contextTop.drawImage(this.overlayImage, 0, 0);
474       }
475       
476       if (this.onFpsUpdate) {
477         var elapsedTime = new Date() - startTime;
478         this.onFpsUpdate(~~(1000 / elapsedTime));
479       }
480       
481       this.fire('after:render');
482       
483       return this;
484     },
485 
486     /**
487      * Method to render only the top canvas.
488      * Also used to render the group selection box.
489      * @method renderTop
490      * @return {fabric.Canvas} thisArg
491      * @chainable
492      */
493     renderTop: function () {
494       this.clearContext(this.contextTop || this.contextContainer);
495       
496       if (this.overlayImage) {
497         (this.contextTop || this.contextContainer).drawImage(this.overlayImage, 0, 0);
498       }
499       
500       // we render the top context - last object
501       if (this.selection && this._groupSelector) {
502         this._drawSelection();
503       }
504       
505       // delegate rendering to group selection if one exists
506       // used for drawing selection borders/corners
507       var activeGroup = this.getActiveGroup();
508       if (activeGroup) {
509         activeGroup.render(this.contextTop);
510       }
511       
512       this.fire('after:render');
513       
514       return this;
515     },

530     /**
531      * Exports canvas element to a dataurl image (allowing to change image size via multiplier).
532      * @method toDataURLWithMultiplier
533      * @param {String} format (png|jpeg)
534      * @param {Number} multiplier
535      * @return {String}
536      */
537     toDataURLWithMultiplier: function (format, multiplier) {
538       
539       var origWidth = this.getWidth(),
540           origHeight = this.getHeight(),
541           scaledWidth = origWidth * multiplier,
542           scaledHeight = origHeight * multiplier,
543           activeObject = this.getActiveObject();
544       
545       this.setWidth(scaledWidth).setHeight(scaledHeight);
546       this.contextTop.scale(multiplier, multiplier);
547       
548       if (activeObject) {
549         this.deactivateAll();
550       }
551       
552       // restoring width, height for `renderAll` to draw 
553       // background properly (while context is scaled)
554       this.width = origWidth;
555       this.height = origHeight;
556       
557       this.renderAll(true);
558       
559       var dataURL = this.toDataURL(format);
560 
561       this.contextTop.scale(1 / multiplier,  1 / multiplier);
562       this.setWidth(origWidth).setHeight(origHeight);
563       
564       if (activeObject) {
565         this.setActiveObject(activeObject);
566       }
567       this.renderAll();
568       
569       return dataURL;
570     },

733     /**
734      * Moves an object one level down in stack of drawn objects
735      * @method sendBackwards
736      * @param object {fabric.Object} Object to send
737      * @return {fabric.Canvas} thisArg
738      * @chainable
739      */
740     sendBackwards: function (object) {
741       var idx = this._objects.indexOf(object),
742           nextIntersectingIdx = idx;
743       
744       // if object is not on the bottom of stack
745       if (idx !== 0) {
746         
747         // traverse down the stack looking for the nearest intersecting object
748         for (var i=idx-1; i>=0; --i) {
749           if (object.intersectsWithObject(this._objects[i]) || object.isContainedWithinObject(this._objects[i])) {
750             nextIntersectingIdx = i;
751             break;
752           }
753         }
754         removeFromArray(this._objects, object);
755         this._objects.splice(nextIntersectingIdx, 0, object);
756       }
757       return this.renderAll();
758     },

760     /**
761      * Moves an object one level up in stack of drawn objects
762      * @method sendForward
763      * @param object {fabric.Object} Object to send
764      * @return {fabric.Canvas} thisArg
765      * @chainable
766      */
767     bringForward: function (object) {
768       var objects = this.getObjects(),
769           idx = objects.indexOf(object),
770           nextIntersectingIdx = idx;
771 
772       
773       // if object is not on top of stack (last item in an array)
774       if (idx !== objects.length-1) {
775         
776         // traverse up the stack looking for the nearest intersecting object
777         for (var i = idx + 1, l = this._objects.length; i < l; ++i) {
778           if (object.intersectsWithObject(objects[i]) || object.isContainedWithinObject(this._objects[i])) {
779             nextIntersectingIdx = i;
780             break;
781           }
782         }
783         removeFromArray(objects, object);
784         objects.splice(nextIntersectingIdx, 0, object);
785       }
786       this.renderAll();
787     },

817       var objects = this.getObjects(),
