  4    * Cross-browser wrapper for setting element's style
  5    * @method setStyle
  6    * @memberOf fabric.util
  7    * @param {HTMLElement} element
  8    * @param {Object} styles
  9    * @return {HTMLElement} Element that was passed as a first argument
 10    */
 11   function setStyle(element, styles) {
 12     var elementStyle = element.style, match;
 13     if (!elementStyle) {
 14       return element;
 15     }
 16     if (typeof styles === 'string') {
 17       element.style.cssText += ';' + styles;
 18       return styles.indexOf('opacity') > -1
 19         ? setOpacity(element, styles.match(/opacity:\s*(\d?\.?\d*)/)[1])
 20         : element;
 21     }
 22     for (var property in styles) {
 23       if (property === 'opacity') {
 24         setOpacity(element, styles[property]);
 25       }
 26       else {
 27         var normalizedProperty = (property === 'float' || property === 'cssFloat')
 28           ? (typeof elementStyle.styleFloat === 'undefined' ? 'cssFloat' : 'styleFloat')
 29           : property;
 30         elementStyle[normalizedProperty] = styles[property];
 31       }
 32     }
 33     return element;
 34   }
