1 (function(){
  2   
  3   /**
  4    * Copies all enumerable properties of one object to another
  5    * @memberOf fabric.util.object
  6    * @method extend
  7    * @param {Object} destination Where to copy to
  8    * @param {Object} source Where to copy from
  9    */
 10   function extend(destination, source) {
 11     // JScript DontEnum bug is not taken care of
 12     for (var property in source) {
 13       destination[property] = source[property];
 14     }
 15     return destination;
 16   }
 17 
 18   /**
 19    * Creates an empty object and copies all enumerable properties of another object to it
 20    * @method clone
 21    * @memberOf fabric.util.object
 22    * @param {Object} object Object to clone
 23    */
 24   function clone(object) {
 25     return extend({ }, object);
 26   }
 27 
 28   /** @namespace fabric.util.object */
 29   fabric.util.object = {
 30     extend: extend,
 31     clone: clone
 32   };
 33   
 34 })();
