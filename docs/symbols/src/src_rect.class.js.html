 15    * @class Rect
 16    * @extends fabric.Object
 17    */
 18   fabric.Rect = fabric.util.createClass(fabric.Object, /** @scope fabric.Rect.prototype */ {
 19     
 20     /**
 21      * @property
 22      * @type String
 23      */
 24     type: 'rect',
 25     
 26     /**
 27      * @property
 28      * @type Object
 29      */
 30     options: {
 31       rx: 0,
 32       ry: 0
 33     },
 34     
 35     /**
 36      * Constructor
 37      * @method initialize
 38      * @param options {Object} options object
 39      * @return {Object} thisArg
 40      */
 41     initialize: function(options) {
 42       this._initStateProperties();
 43       this.callSuper('initialize', options);
 44       this._initRxRy();
 45     },
 46     
 47     /**
 48      * Creates `stateProperties` list on an instance, and adds `fabric.Rect` -specific ones to it 
 49      * (such as "rx", "ry", etc.)
 50      * @private
 51      * @method _initStateProperties
 52      */
 53     _initStateProperties: function() {
 54       this.stateProperties = this.stateProperties.concat(['rx', 'ry']);
 55     },
 56     
 57     /**
 58      * @private
 59      * @method _initRxRy
 60      */
 61     _initRxRy: function() {
 62       if (this.rx && !this.ry) {
 63         this.ry = this.rx;
 64       }
 65       else if (this.ry && !this.rx) {
 66         this.rx = this.ry;
 67       }
 68     },
 69     
 70     /**
 71      * @private
 72      * @method _render
 73      * @param ctx {CanvasRenderingContext2D} context to render on
 74      */
 75     _render: function(ctx) {   
 76       var rx = this.rx || 0,
 77           ry = this.ry || 0,
 78           x = -this.width / 2,
 79           y = -this.height / 2,
 80           w = this.width,
 81           h = this.height;
 82       
 83       ctx.beginPath();
 84       ctx.globalAlpha *= this.opacity;
 85       
 86       if (this.group) {
 87         ctx.translate(this.x || 0, this.y || 0);
 88       }
 89       
 90       ctx.moveTo(x+rx, y);
 91       ctx.lineTo(x+w-rx, y);
 92       ctx.bezierCurveTo(x+w, y, x+w, y+ry, x+w, y+ry);
 93       ctx.lineTo(x+w, y+h-ry);
 94       ctx.bezierCurveTo(x+w,y+h,x+w-rx,y+h,x+w-rx,y+h);
 95       ctx.lineTo(x+rx,y+h);
 96       ctx.bezierCurveTo(x,y+h,x,y+h-ry,x,y+h-ry);
 97       ctx.lineTo(x,y+ry);
 98       ctx.bezierCurveTo(x,y,x+rx,y,x+rx,y);
 99       ctx.closePath();
100       
101       if (this.fill) {
102         ctx.fill();
103       }
104       if (this.stroke) {
105         ctx.stroke();
106       }
107     },
108     
109     // since our coordinate system differs from that of SVG
110     _normalizeLeftTopProperties: function(parsedAttributes) {
111       if (parsedAttributes.left) {
112         this.set('left', parsedAttributes.left + this.getWidth() / 2);
113       }
114       this.set('x', parsedAttributes.left || 0);
115       if (parsedAttributes.top) {
116         this.set('top', parsedAttributes.top + this.getHeight() / 2);
117       }
118       this.set('y', parsedAttributes.top || 0);
119       return this;
120     },
121     
122     /**
123      * @method complexity
124      * @return {Number} complexity
125      */
126     complexity: function() {
127       return 1;
128     },
129     
130     /**
131      * Returns svg representation of an instance
132      * @method toSVG
133      * @return {string} svg representation of an instance
134      */
135     toSVG: function() {
136       return '<rect ' +
137               'x="' + (-1 * this.width / 2) + '" y="' + (-1 * this.height / 2) + '" ' +
138               'width="' + this.width + '" height="' + this.height + '" ' +
139               'style="' + this.getSvgStyles() + '" ' +
140               'transform="' + this.getSvgTransform() + '" ' +
141               '/>';
142     }
143   });

148    * List of attribute names to account for when parsing SVG element (used by `fabric.Rect.fromElement`)
149    * @static

151   fabric.Rect.ATTRIBUTE_NAMES = 'x y width height rx ry fill fill-opacity opacity stroke stroke-width transform'.split(' ');

153   /**
154    * @private
155    */
156   function _setDefaultLeftTopValues(attributes) {
157     attributes.left = attributes.left || 0;
158     attributes.top  = attributes.top  || 0;
159     return attributes;
160   }

162   /**
163    * Returns fabric.Rect instance from an SVG element
164    * @static
165    * @method fabric.Rect.fromElement
166    * @param element {SVGElement} element to parse
167    * @param options {Object} options object
168    * @return {fabric.Rect} instance of fabric.Rect
169    */
170   fabric.Rect.fromElement = function(element, options) {
171     if (!element) {
172       return null;
173     }
174     
175     var parsedAttributes = fabric.parseAttributes(element, fabric.Rect.ATTRIBUTE_NAMES);
176     parsedAttributes = _setDefaultLeftTopValues(parsedAttributes);
177     
178     var rect = new fabric.Rect(fabric.util.object.extend((options ? fabric.util.object.clone(options) : { }), parsedAttributes));
179     rect._normalizeLeftTopProperties(parsedAttributes);
180     
181     return rect;
182   };

184   /**
185    * Returns fabric.Rect instance from an object representation
186    * @static
187    * @method fabric.Rect.fromObject
188    * @param object {Object} object to create an instance from
189    * @return {Object} instance of fabric.Rect
190    */
191   fabric.Rect.fromObject = function(object) {
192     return new fabric.Rect(object);
