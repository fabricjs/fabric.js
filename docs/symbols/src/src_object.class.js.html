478     /**

486       this.currentWidth = this.width * this.scaleX;

489       this._hypotenuse = Math.sqrt(
490         Math.pow(this.currentWidth / 2, 2) +
491         Math.pow(this.currentHeight / 2, 2));
492 
493       this._angle = Math.atan(this.currentHeight / this.currentWidth);
494 
495       // offset added for rotate and scale actions

497           offsetY = Math.sin(this._angle + this.theta) * this._hypotenuse,
498           theta = this.theta,
499           sinTh = Math.sin(theta),
500           cosTh = Math.cos(theta);

502       var tl = {

506       var tr = {
507         x: tl.x + (this.currentWidth * cosTh),
508         y: tl.y + (this.currentWidth * sinTh)
509       };
510       var br = {

512         y: tr.y + (this.currentHeight * cosTh)

566     /**
567      * Draws borders of an object's bounding box.
568      * Requires public properties: width, height
569      * Requires public options: padding, borderColor
570      * @method drawBorders
571      * @param {CanvasRenderingContext2D} ctx Context to draw on
572      * @return {fabric.Object} thisArg
573      * @chainable
574      */
575     drawBorders: function(ctx) {
576       if (!this.hasBorders) return;
577 
578       var padding = this.padding,
579           padding2 = padding * 2;
580 
581       ctx.save();
582 
583       ctx.globalAlpha = this.isMoving ? this.borderOpacityWhenMoving : 1;
584       ctx.strokeStyle = this.borderColor;
585 
586       var scaleX = 1 / (this.scaleX < this.MIN_SCALE_LIMIT ? this.MIN_SCALE_LIMIT : this.scaleX),
587           scaleY = 1 / (this.scaleY < this.MIN_SCALE_LIMIT ? this.MIN_SCALE_LIMIT : this.scaleY);
588 
589       ctx.lineWidth = 1 / this.borderScaleFactor;
590 
591       ctx.scale(scaleX, scaleY);
592 
593       var w = this.getWidth(),
594           h = this.getHeight();
595 
596       ctx.strokeRect(
597         ~~(-(w / 2) - padding) + 0.5, // offset needed to make lines look sharper
598         ~~(-(h / 2) - padding) + 0.5,
599         ~~(w + padding2),
600         ~~(h + padding2)
601       );
602 
603       if (this.hasRotatingPoint && !this.hideCorners && !this.lockRotation) {
604         var rotateHeight = (-h/2);
605         var rotateWidth = (-w/2);
606 
607         ctx.beginPath();
608         ctx.moveTo(0, rotateHeight);
609         ctx.lineTo(0, rotateHeight - 40);
610         ctx.closePath();
611         ctx.stroke();
612       }
613 
614       ctx.restore();
615       return this;
616     },

618     /**
619      * Draws corners of an object's bounding box.
620      * Requires public properties: width, height, scaleX, scaleY
621      * Requires public options: cornersize, padding
622      * @method drawCorners
623      * @param {CanvasRenderingContext2D} ctx Context to draw on
624      * @return {fabric.Object} thisArg
625      * @chainable
626      */
627     drawCorners: function(ctx) {
628       if (!this.hasControls) return;
629 
630       var size = this.cornersize,
631           size2 = size / 2,
632           padding = this.padding,
633           left = -(this.width / 2),
634           top = -(this.height / 2),
635           _left,
636           _top,
637           sizeX = size / this.scaleX,
638           sizeY = size / this.scaleY,
639           scaleOffsetY = (padding + size2) / this.scaleY,
640           scaleOffsetX = (padding + size2) / this.scaleX,
641           scaleOffsetSizeX = (padding + size2 - size) / this.scaleX,
642           scaleOffsetSizeY = (padding + size2 - size) / this.scaleY,
643           height = this.height;
644 
645       ctx.save();
646 
647       ctx.globalAlpha = this.isMoving ? this.borderOpacityWhenMoving : 1;
648       ctx.fillStyle = this.cornerColor;
649 
650       // top-left
651       _left = left - scaleOffsetX;
652       _top = top - scaleOffsetY;
653       ctx.fillRect(_left, _top, sizeX, sizeY);
654 
655       // top-right
656       _left = left + this.width - scaleOffsetX;
657       _top = top - scaleOffsetY;
658       ctx.fillRect(_left, _top, sizeX, sizeY);
659 
660       // bottom-left
661       _left = left - scaleOffsetX;
662       _top = top + height + scaleOffsetSizeY;
663       ctx.fillRect(_left, _top, sizeX, sizeY);
664 
665       // bottom-right
666       _left = left + this.width + scaleOffsetSizeX;
667       _top = top + height + scaleOffsetSizeY;
668       ctx.fillRect(_left, _top, sizeX, sizeY);
669 
670       // middle-top
671       _left = left + this.width/2 - scaleOffsetX;

674 
675       // middle-bottom
676       _left = left + this.width/2 - scaleOffsetX;
677       _top = top + height + scaleOffsetSizeY;
678       ctx.fillRect(_left, _top, sizeX, sizeY);
679 
680       // middle-right
681       _left = left + this.width + scaleOffsetSizeX;
682       _top = top + height/2 - scaleOffsetY;
683       ctx.fillRect(_left, _top, sizeX, sizeY);
684 
685       // middle-left
686       _left = left - scaleOffsetX;
687       _top = top + height/2 - scaleOffsetY;
688       ctx.fillRect(_left, _top, sizeX, sizeY);
689

695         // ctx.save();

701         _top = top - (45 / this.scaleY);
702         ctx.fillRect(_left, _top, sizeX, sizeY);

961     /**
962      * Helper method to determine how many cross points are between the 4 image edges
963      * and the horizontal line determined by the position of our mouse when clicked on canvas
964      * @method _findCrossPoints
965      * @private
966      * @param ex {Number} x coordinate of the mouse
967      * @param ey {Number} y coordinate of the mouse
968      * @param oCoords {Object} Coordinates of the image being evaluated
969      */
970     _findCrossPoints: function(ex, ey, oCoords) {
971       var b1, b2, a1, a2, xi, yi,
972           xcount = 0,
973           iLine;
974 
975       for (var lineKey in oCoords) {
976         iLine = oCoords[lineKey];
977         // optimisation 1: line below dot. no cross
978         if ((iLine.o.y < ey) && (iLine.d.y < ey)) {
979           continue;
980         }
981         // optimisation 2: line above dot. no cross
982         if ((iLine.o.y >= ey) && (iLine.d.y >= ey)) {
983           continue;
984         }
985         // optimisation 3: vertical line case
986         if ((iLine.o.x == iLine.d.x) && (iLine.o.x >= ex)) {
987           xi = iLine.o.x;
988           yi = ey;
989         }
990         // calculate the intersection point
991         else {
992           b1 = 0;
993           b2 = (iLine.d.y-iLine.o.y)/(iLine.d.x-iLine.o.x);
994           a1 = ey-b1*ex;
995           a2 = iLine.o.y-b2*iLine.o.x;
996 
997           xi = - (a1-a2)/(b1-b2);
998           yi = a1+b1*xi;
999         }
1000         // dont count xi < ex cases
1001         if (xi >= ex) {
1002           xcount += 1;
1003         }
1004         // optimisation 4: specific for square images
1005         if (xcount == 2) {
1006           break;
1007         }
1008       }
1009       return xcount;
1010     },
