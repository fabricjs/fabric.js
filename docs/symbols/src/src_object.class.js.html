350     /**

352      * @param {CanvasRenderingContext2D} ctx context to render on

356

358       if (this.width === 0 || this.height === 0) return;
359

361 
362       var m = this.transformMatrix;
363       if (m) {

373         ctx.strokeStyle = this.stroke;
374       }
375 
376       if (this.overlayFill) {
377         ctx.fillStyle = this.overlayFill;
378       }
379       else if (this.fill) {
380         ctx.fillStyle = this.fill;
381       }

389       // }
390       this._render(ctx, noTransform);
391 
392       if (this.active && !noTransform) {
393         this.drawBorders(ctx);

485     /**
486      * Sets corner position coordinates based on current angle, width and height.
487      * @method setCoords
488      * return {fabric.Object} thisArg
489      * @chainable
490      */
491     setCoords: function() {
492 
493       this.currentWidth = this.width * this.scaleX;
494       this.currentHeight = this.height * this.scaleY;
495 
496       this._hypotenuse = Math.sqrt(
497         Math.pow(this.currentWidth / 2, 2) +
498         Math.pow(this.currentHeight / 2, 2));
499 
500       this._angle = Math.atan(this.currentHeight / this.currentWidth);
501 
502       // offset added for rotate and scale actions
503       var offsetX = Math.cos(this._angle + this.theta) * this._hypotenuse,
504           offsetY = Math.sin(this._angle + this.theta) * this._hypotenuse,
505           theta = this.theta,
506           sinTh = Math.sin(theta),
507           cosTh = Math.cos(theta);
508 
509       var tl = {
510         x: this.left - offsetX,
511         y: this.top - offsetY
512       };
513       var tr = {
514         x: tl.x + (this.currentWidth * cosTh),
515         y: tl.y + (this.currentWidth * sinTh)
516       };
517       var br = {
518         x: tr.x - (this.currentHeight * sinTh),
519         y: tr.y + (this.currentHeight * cosTh)
520       };
521       var bl = {
522         x: tl.x - (this.currentHeight * sinTh),
523         y: tl.y + (this.currentHeight * cosTh)
524       };
525       var ml = {
526         x: tl.x - (this.currentHeight/2 * sinTh),
527         y: tl.y + (this.currentHeight/2 * cosTh)
528       };
529       var mt = {
530         x: tl.x + (this.currentWidth/2 * cosTh),
531         y: tl.y + (this.currentWidth/2 * sinTh)
532       };
533       var mr = {
534         x: tr.x - (this.currentHeight/2 * sinTh),
535         y: tr.y + (this.currentHeight/2 * cosTh)
536       };
537       var mb = {
538         x: bl.x + (this.currentWidth/2 * cosTh),
539         y: bl.y + (this.currentWidth/2 * sinTh)
540       };
541       var mtr = {
542         x: tl.x + (this.currentWidth/2 * cosTh),
543         y: tl.y + (this.currentWidth/2 * sinTh)
544       };
545 
546       // debugging
547 
548       // setTimeout(function() {
549       //         canvas.contextTop.fillStyle = 'green';
550       //         canvas.contextTop.fillRect(mb.x, mb.y, 3, 3);
551       //         canvas.contextTop.fillRect(bl.x, bl.y, 3, 3);
552       //         canvas.contextTop.fillRect(br.x, br.y, 3, 3);
553       //         canvas.contextTop.fillRect(tl.x, tl.y, 3, 3);
554       //         canvas.contextTop.fillRect(tr.x, tr.y, 3, 3);
555       //         canvas.contextTop.fillRect(ml.x, ml.y, 3, 3);
556       //         canvas.contextTop.fillRect(mr.x, mr.y, 3, 3);
557       //         canvas.contextTop.fillRect(mt.x, mt.y, 3, 3);
558       //       }, 50);
559 
560       // clockwise
561       this.oCoords = { tl: tl, tr: tr, br: br, bl: bl, ml: ml, mt: mt, mr: mr, mb: mb, mtr: mtr };
562 
563       // set coordinates of the draggable boxes in the corners used to scale/rotate the image

593     /**
594      * Draws borders of an object's bounding box.
595      * Requires public properties: width, height
596      * Requires public options: padding, borderColor
597      * @method drawBorders
598      * @param {CanvasRenderingContext2D} ctx Context to draw on
599      * @return {fabric.Object} thisArg
600      * @chainable
601      */
602     drawBorders: function(ctx) {
603       if (!this.hasBorders) return;
604 
605       var padding = this.padding,
606           padding2 = padding * 2;
607 
608       ctx.save();
609 
610       ctx.globalAlpha = this.isMoving ? this.borderOpacityWhenMoving : 1;
611       ctx.strokeStyle = this.borderColor;
612 
613       var scaleX = 1 / (this.scaleX < this.MIN_SCALE_LIMIT ? this.MIN_SCALE_LIMIT : this.scaleX),
614           scaleY = 1 / (this.scaleY < this.MIN_SCALE_LIMIT ? this.MIN_SCALE_LIMIT : this.scaleY);
615 
616       ctx.lineWidth = 1 / this.borderScaleFactor;
617 
618       ctx.scale(scaleX, scaleY);
619 
620       var w = this.getWidth(),
621           h = this.getHeight();
622 
623       ctx.strokeRect(
624         ~~(-(w / 2) - padding) + 0.5, // offset needed to make lines look sharper
625         ~~(-(h / 2) - padding) + 0.5,
626         ~~(w + padding2),
627         ~~(h + padding2)
628       );
629 
630       if (this.hasRotatingPoint && !this.hideCorners && !this.lockRotation) {
631         var rotateHeight = (this.flipY ? h : -h) / 2;
632         var rotateWidth = (-w/2);
633 
634         ctx.beginPath();
635         ctx.moveTo(0, rotateHeight);
636         ctx.lineTo(0, rotateHeight + (this.flipY ? this.rotatingPointOffset : -this.rotatingPointOffset));
637         ctx.closePath();
638         ctx.stroke();
639       }
640 
641       ctx.restore();
642       return this;
643     },

645     /**
646      * Draws corners of an object's bounding box.
647      * Requires public properties: width, height, scaleX, scaleY
648      * Requires public options: cornersize, padding
649      * @method drawCorners
650      * @param {CanvasRenderingContext2D} ctx Context to draw on
651      * @return {fabric.Object} thisArg
652      * @chainable
653      */
654     drawCorners: function(ctx) {
655       if (!this.hasControls) return;
656 
657       var size = this.cornersize,
658           size2 = size / 2,
659           padding = this.padding,
660           left = -(this.width / 2),
661           top = -(this.height / 2),
662           _left,
663           _top,
664           sizeX = size / this.scaleX,
665           sizeY = size / this.scaleY,
666           scaleOffsetY = (padding + size2) / this.scaleY,
667           scaleOffsetX = (padding + size2) / this.scaleX,
668           scaleOffsetSizeX = (padding + size2 - size) / this.scaleX,
669           scaleOffsetSizeY = (padding + size2 - size) / this.scaleY,
670           height = this.height;
671 
672       ctx.save();
673 
674       ctx.globalAlpha = this.isMoving ? this.borderOpacityWhenMoving : 1;
675       ctx.fillStyle = this.cornerColor;
676 
677       // top-left
678       _left = left - scaleOffsetX;
679       _top = top - scaleOffsetY;
680       ctx.fillRect(_left, _top, sizeX, sizeY);
681 
682       // top-right
683       _left = left + this.width - scaleOffsetX;
684       _top = top - scaleOffsetY;
685       ctx.fillRect(_left, _top, sizeX, sizeY);
686 
687       // bottom-left
688       _left = left - scaleOffsetX;
689       _top = top + height + scaleOffsetSizeY;
690       ctx.fillRect(_left, _top, sizeX, sizeY);
691 
692       // bottom-right
693       _left = left + this.width + scaleOffsetSizeX;
694       _top = top + height + scaleOffsetSizeY;
695       ctx.fillRect(_left, _top, sizeX, sizeY);
696 
697       // middle-top
698       _left = left + this.width/2 - scaleOffsetX;
699       _top = top - scaleOffsetY;
700       ctx.fillRect(_left, _top, sizeX, sizeY);
701 
702       // middle-bottom
703       _left = left + this.width/2 - scaleOffsetX;
704       _top = top + height + scaleOffsetSizeY;
705       ctx.fillRect(_left, _top, sizeX, sizeY);
706 
707       // middle-right
708       _left = left + this.width + scaleOffsetSizeX;
709       _top = top + height/2 - scaleOffsetY;
710       ctx.fillRect(_left, _top, sizeX, sizeY);
711 
712       // middle-left
713       _left = left - scaleOffsetX;
714       _top = top + height/2 - scaleOffsetY;
715       ctx.fillRect(_left, _top, sizeX, sizeY);
716 
717       // middle-top-rotate
718       if (this.hasRotatingPoint) {
719         // _left = left + this.width/2;
720         // _top = top - (45 / this.scaleY) + scaleOffsetY;
721 
722         // ctx.save();
723         // ctx.beginPath();
724         // ctx.arc(_left, _top, sizeX / 2, 0, Math.PI * 2, false);
725         // ctx.fill();
726         // ctx.restore();
727 
728         _left = left + this.width/2 - scaleOffsetX;
729 
730         _top = this.flipY ?
731           (top + height + (this.rotatingPointOffset / this.scaleY) - sizeY/2)
732           : (top - (this.rotatingPointOffset / this.scaleY) - sizeY/2);
733 
734         ctx.fillRect(_left, _top, sizeX, sizeY);
735       }
736 
737       ctx.restore();
738 
739       return this;
740     },

996     /**
997      * Helper method to determine how many cross points are between the 4 image edges
998      * and the horizontal line determined by the position of our mouse when clicked on canvas
999      * @method _findCrossPoints
1000      * @private
1001      * @param ex {Number} x coordinate of the mouse
1002      * @param ey {Number} y coordinate of the mouse
1003      * @param oCoords {Object} Coordinates of the image being evaluated
1004      */
1005     _findCrossPoints: function(ex, ey, oCoords) {
1006       var b1, b2, a1, a2, xi, yi,
1007           xcount = 0,
1008           iLine;
1009 
1010       for (var lineKey in oCoords) {
1011         iLine = oCoords[lineKey];
1012         // optimisation 1: line below dot. no cross
1013         if ((iLine.o.y < ey) && (iLine.d.y < ey)) {
1014           continue;
1015         }
1016         // optimisation 2: line above dot. no cross
1017         if ((iLine.o.y >= ey) && (iLine.d.y >= ey)) {
1018           continue;
1019         }
1020         // optimisation 3: vertical line case
1021         if ((iLine.o.x == iLine.d.x) && (iLine.o.x >= ex)) {
1022           xi = iLine.o.x;
1023           yi = ey;
1024         }
1025         // calculate the intersection point
1026         else {
1027           b1 = 0;
1028           b2 = (iLine.d.y-iLine.o.y)/(iLine.d.x-iLine.o.x);
1029           a1 = ey-b1*ex;
1030           a2 = iLine.o.y-b2*iLine.o.x;
1031 
1032           xi = - (a1-a2)/(b1-b2);
1033           yi = a1+b1*xi;
1034         }
1035         // dont count xi < ex cases
1036         if (xi >= ex) {
1037           xcount += 1;
1038         }
1039         // optimisation 4: specific for square images
1040         if (xcount == 2) {
1041           break;
1042         }
1043       }
1044       return xcount;
1045     },
