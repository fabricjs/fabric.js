  3   /**
  4    * Populates canvas with data from the specified dataless JSON
  5    * JSON format must conform to the one of `fabric.Canvas#toDatalessJSON`
  6    * @method loadFromDatalessJSON
  7    * @param {String} json JSON string
  8    * @param {Function} callback Callback, invoked when json is parsed 
  9    *                            and corresponding objects (e.g: fabric.Image) 
 10    *                            are initialized
 11    * @return {fabric.Canvas} instance
 12    * @chainable
 13    */
 14   loadFromDatalessJSON: function (json, callback) {
 15 
 16     if (!json) {
 17       return;
 18     }
 19 
 20     // serialize if it wasn't already
 21     var serialized = (typeof json === 'string')
 22       ? JSON.parse(json)
 23       : json;
 24 
 25     if (!serialized || (serialized && !serialized.objects)) return;
 26 
 27     this.clear();
 28 
 29     // TODO: test this
 30     this.backgroundColor = serialized.background;
 31     this._enlivenDatalessObjects(serialized.objects, callback);
 32   },
 33 
 34   /**
 35    * @method _enlivenDatalessObjects
 36    * @param {Array} objects
 37    * @param {Function} callback
 38    */
 39   _enlivenDatalessObjects: function (objects, callback) {
 40 
 41     /** @ignore */
 42     function onObjectLoaded(object, index) {
 43       _this.insertAt(object, index, true);
 44       object.setCoords();
 45       if (++numLoadedObjects === numTotalObjects) {
 46         callback && callback();
 47       }
 48     }
 49 
 50     var _this = this,
 51         numLoadedObjects = 0,
 52         numTotalObjects = objects.length;
 53 
 54     if (numTotalObjects === 0 && callback) {
 55       callback();
 56     }
 57 
 58     try {
 59       objects.forEach(function (obj, index) {
 60 
 61         var pathProp = obj.paths ? 'paths' : 'path';
 62         var path = obj[pathProp];
 63 
 64         delete obj[pathProp];
 65 
 66         if (typeof path !== 'string') {
 67           switch (obj.type) {
 68             case 'image':
 69             case 'text':
 70               fabric[fabric.util.string.capitalize(obj.type)].fromObject(obj, function (o) {
 71                 onObjectLoaded(o, index);
 72               });
 73               break;
 74             default:
 75               var klass = fabric[fabric.util.string.camelize(fabric.util.string.capitalize(obj.type))];
 76               if (klass && klass.fromObject) {
 77                 // restore path
 78                 if (path) {
 79                   obj[pathProp] = path;
 80                 }
 81                 onObjectLoaded(klass.fromObject(obj), index);
 82               }
 83               break;
 84           }
 85         }
 86         else {
 87           if (obj.type === 'image') {
 88             fabric.util.loadImage(path, function (image) {
 89               var oImg = new fabric.Image(image);
 90               
 91               oImg.setSourcePath(path);
 92 
 93               fabric.util.object.extend(oImg, obj);
 94               oImg.setAngle(obj.angle);
 95 
 96               onObjectLoaded(oImg, index);
 97             });
 98           }
 99           else if (obj.type === 'text') {
100 
101             obj.path = path;
102             var object = fabric.Text.fromObject(obj);
103             var onscriptload = function () {
104               // TODO (kangax): find out why Opera refuses to work without this timeout
105               if (Object.prototype.toString.call(fabric.window.opera) === '[object Opera]') {
106                 setTimeout(function () {
107                   onObjectLoaded(object, index);
108                 }, 500);
109               }
110               else {
111                 onObjectLoaded(object, index);
112               }
113             }
114 
115             fabric.util.getScript(path, onscriptload);
116           }
117           else {
118             fabric.loadSVGFromURL(path, function (elements, options) {
119               if (elements.length > 1) {
120                 var object = new fabric.PathGroup(elements, obj);
121               }
122               else {
123                 var object = elements[0];
124               }
125               object.setSourcePath(path);
126 
127               // copy parameters from serialied json to object (left, top, scaleX, scaleY, etc.)
128               // skip this step if an object is a PathGroup, since we already passed it options object before
129               if (!(object instanceof fabric.PathGroup)) {
130                 fabric.util.object.extend(object, obj);
131                 if (typeof obj.angle !== 'undefined') {
132                   object.setAngle(obj.angle);
133                 }
134               }
135 
136               onObjectLoaded(object, index);
137             });
138           }
139         }
140       }, this);
141     } 
142     catch(e) {
143       fabric.log(e.message);
144     }
145   },

147   /**
148    * Populates canvas with data from the specified JSON
149    * JSON format must conform to the one of `fabric.Canvas#toJSON`
150    * @method loadFromJSON
151    * @param {String} json JSON string
152    * @param {Function} callback Callback, invoked when json is parsed 
153    *                            and corresponding objects (e.g: fabric.Image) 
154    *                            are initialized
155    * @return {fabric.Canvas} instance
156    * @chainable
157    */
158   loadFromJSON: function (json, callback) {
159     if (!json) return;
160     
161     var serialized = JSON.parse(json);
162     if (!serialized || (serialized && !serialized.objects)) return;
163     
164     this.clear();
165     var _this = this;
166     this._enlivenObjects(serialized.objects, function () {
167       _this.backgroundColor = serialized.background;
168       if (callback) {
169         callback();
170       }
171     });
172     
173     return this;
174   },

176   /**
177    * @method _enlivenObjects
178    * @param {Array} objects
179    * @param {Function} callback
180    */
181   _enlivenObjects: function (objects, callback) {
182     var numLoadedImages = 0,
183         // get length of all images 
184         numTotalImages = objects.filter(function (o) {
185           return o.type === 'image';
186         }).length;
187     
188     var _this = this;
189     
190     objects.forEach(function (o, index) {
191       if (!o.type) {
192         return;
193       }
194       switch (o.type) {
195         case 'image':
196         case 'font':
197           fabric[fabric.util.string.capitalize(o.type)].fromObject(o, function (o) {
198             _this.insertAt(o, index, true);
199             if (++numLoadedImages === numTotalImages) {
200               if (callback) {
201                 callback();
202               }
203             }
204           });
205           break;
206         default:
207           var klass = fabric[fabric.util.string.camelize(fabric.util.string.capitalize(o.type))];
208           if (klass && klass.fromObject) {
209             _this.insertAt(klass.fromObject(o), index, true);
210           }
211           break;
212       }
213     });
214     
215     if (numTotalImages === 0 && callback) {
216       callback();
217     }
218   },

245   /**
246    * Clones canvas instance
247    * @method clone
248    * @param {Object} [callback] Expects `onBeforeClone` and `onAfterClone` functions
249    * @return {fabric.Canvas} Clone of this instance
250    */
251   clone: function (callback) {
252     var el = fabric.document.createElement('canvas');
253     
254     el.width = this.getWidth();
255     el.height = this.getHeight();
256         
257     // cache
258     var clone = this.__clone || (this.__clone = new fabric.Canvas(el));
259     clone.clipTo = this.clipTo;
260     
261     return clone.loadFromJSON(JSON.stringify(this.toJSON()), function () {
262       if (callback) {
263         callback(clone);
264       }
265     });
