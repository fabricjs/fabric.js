 12   /**
 13    * The purpose of {@link fabric.Color} is to abstract and encapsulate common color operations;
 14    * {@link fabric.Color} is a constructor and creates instances of {@link fabric.Color} objects.
 15    *
 16    * @class Color
 17    * @memberOf fabric
 18    * @param {String} color (optional) in hex or rgb(a) format
 19    */
 20   function Color(color) {
 21     if (!color) {
 22       this.setSource([0, 0, 0, 1]);
 23     }
 24     else {
 25       this._tryParsingColor(color);
 26     }
 27   }
 28   
 29   fabric.Color = Color;
 30   
 31   fabric.Color.prototype = /** @scope fabric.Color.prototype */ {
 32     
 33     /**
 34      * @private
 35      * @method _tryParsingColor
 36      */
 37     _tryParsingColor: function(color) {
 38       var source = Color.sourceFromHex(color);
 39       if (!source) {
 40         source = Color.sourceFromRgb(color);
 41       }
 42       if (source) {
 43         this.setSource(source);
 44       }
 45     },

 47     /**
 48      * Returns source of this color (where source is an array representation; ex: [200, 200, 100, 1])
 49      * @method getSource
 50      * @return {Array}
 51      */
 52     getSource: function() {
 53       return this._source;
 54     },
 55 
 56     /**
 57      * Sets source of this color (where source is an array representation; ex: [200, 200, 100, 1])
 58      * @method setSource
 59      * @param {Array} source
 60      */
 61     setSource: function(source) {
 62       this._source = source;
 63     },
 64 
 65     /**
 66      * Returns color represenation in RGB format
 67      * @method toRgb
 68      * @return {String} ex: rgb(0-255,0-255,0-255)
 69      */
 70     toRgb: function() {
 71       var source = this.getSource();
 72       return 'rgb(' + source[0] + ',' + source[1] + ',' + source[2] + ')';
 73     },
 74 
 75     /**
 76      * Returns color represenation in RGBA format
 77      * @method toRgba
 78      * @return {String} ex: rgba(0-255,0-255,0-255,0-1)
 79      */
 80     toRgba: function() {
 81       var source = this.getSource();
 82       return 'rgba(' + source[0] + ',' + source[1] + ',' + source[2] + ',' + source[3] + ')';
 83     },
 84 
 85     /**
 86      * Returns color represenation in HEX format
 87      * @method toHex
 88      * @return {String} ex: FF5555
 89      */
 90     toHex: function() {
 91       var source = this.getSource();
 92 
 93       var r = source[0].toString(16);
 94       r = (r.length == 1) ? ('0' + r) : r;
 95 
 96       var g = source[1].toString(16);
 97       g = (g.length == 1) ? ('0' + g) : g;
 98 
 99       var b = source[2].toString(16);
100       b = (b.length == 1) ? ('0' + b) : b;
101 
102       return r.toUpperCase() + g.toUpperCase() + b.toUpperCase();
103     },
104 
105     /**
106      * Gets value of alpha channel for this color 
107      * @method getAlpha
108      * @return {Number} 0-1
109      */
110     getAlpha: function() {
111       return this.getSource()[3];
112     },
113 
114     /**
115      * Sets value of alpha channel for this color
116      * @method setAlpha
117      * @param {Number} 0-1
118      * @return {fabric.Color} thisArg
119      */
120     setAlpha: function(alpha) {
121       var source = this.getSource();
122       source[3] = alpha;
123       this.setSource(source);
124       return this;
125     },
126 
127     /**
128      * Transforms color to its grayscale representation
129      * @method toGrayscale
130      * @return {fabric.Color} thisArg
131      */
132     toGrayscale: function() {
133       var source = this.getSource(),
134           average = parseInt((source[0] * 0.3 + source[1] * 0.59 + source[2] * 0.11).toFixed(0), 10),
135           currentAlpha = source[3];
136       this.setSource([average, average, average, currentAlpha]);
137       return this;
138     },
139 
140     /**
141      * Transforms color to its black and white representation
142      * @method toGrayscale
143      * @return {fabric.Color} thisArg
144      */
145     toBlackWhite: function(threshold) {
146       var source = this.getSource(),
147           average = (source[0] * 0.3 + source[1] * 0.59 + source[2] * 0.11).toFixed(0),
148           currentAlpha = source[3],
149           threshold = threshold || 127;
150 
151       average = (Number(average) < Number(threshold)) ? 0 : 255;
152       this.setSource([average, average, average, currentAlpha]);
153       return this;
154     },
155 
156     /**
157      * Overlays color with another color
158      * @method overlayWith
159      * @param {String|fabric.Color} otherColor
160      * @return {fabric.Color} thisArg
161      */
162     overlayWith: function(otherColor) {
163       if (!(otherColor instanceof Color)) {
164         otherColor = new Color(otherColor);
165       }
166 
167       var result = [],
168           alpha = this.getAlpha(),
169           otherAlpha = 0.5,
170           source = this.getSource(),
171           otherSource = otherColor.getSource();
172 
173       for (var i = 0; i < 3; i++) {
174         result.push(Math.round((source[i] * (1 - otherAlpha)) + (otherSource[i] * otherAlpha)));
175       }
176 
177       result[3] = alpha;
178       this.setSource(result);
179       return this;
180     }

183   /**
184    * Regex matching color in RGB or RGBA formats (ex: rgb(0, 0, 0), rgb(255, 100, 10, 0.5), rgb(1,1,1))
185    * @static
186    * @field
187    */
188   fabric.Color.reRGBa = /^rgba?\((\d{1,3})\s*,\s*(\d{1,3})\s*,\s*(\d{1,3})(?:\s*,\s*(\d+(?:\.\d+)?))?\)$/;
189   
190   /**
191    * Regex matching color in HEX format (ex: #FF5555, 010155, aff)
192    * @static
193    * @field
194    */
195   fabric.Color.reHex = /^#?([0-9a-f]{6}|[0-9a-f]{3})$/i;

197   /**
198    * Returns new color object, when given a color in RGB format
199    * @method fromRgb
200    * @param {String} color ex: rgb(0-255,0-255,0-255)
201    * @return {fabric.Color}
202    */
203   fabric.Color.fromRgb = function(color) {
204     return Color.fromSource(Color.sourceFromRgb(color));
205   };
206   
207   /**
208    * Returns array represenatation (ex: [100, 100, 200, 1]) of a color that's in RGB or RGBA format
209    * @method sourceFromRgb
210    * @param {String} color ex: rgb(0-255,0-255,0-255)
211    * @return {Array} source
212    */
213   fabric.Color.sourceFromRgb = function(color) {
214     var match = color.match(Color.reRGBa);
215     if (match) {
216       return [
217         parseInt(match[1], 10),
218         parseInt(match[2], 10),
219         parseInt(match[3], 10),
220         match[4] ? parseFloat(match[4]) : 1
221       ];
222     }
223   };
224

226    * Returns new color object, when given a color in RGBA format
227    * @static
228    * @function
229    * @method fromRgba
230    * @param {String} color
231    * @return {fabric.Color}
232    */
233   fabric.Color.fromRgba = Color.fromRgb;

245   /**
246    * Returns array represenatation (ex: [100, 100, 200, 1]) of a color that's in HEX format
247    * @static
248    * @method sourceFromHex
249    * @param {String} color ex: FF5555
250    * @return {Array} source
251    */
252   fabric.Color.sourceFromHex = function(color) {
253     if (color.match(Color.reHex)) {
254       var value = color.slice(color.indexOf('#') + 1),
255           isShortNotation = (value.length === 3),
256           r = isShortNotation ? (value.charAt(0) + value.charAt(0)) : value.substring(0, 2),
257           g = isShortNotation ? (value.charAt(1) + value.charAt(1)) : value.substring(2, 4),
258           b = isShortNotation ? (value.charAt(2) + value.charAt(2)) : value.substring(4, 6);
259 
260       return [
261         parseInt(r, 16),
262         parseInt(g, 16),
263         parseInt(b, 16),
264         1
265       ];
266     }
267   };

277     oColor.setSource(source);
278     return oColor;
