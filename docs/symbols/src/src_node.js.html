 14   function request(url, encoding, callback) {
 15     var oURL = URL.parse(url),
 16         client = HTTP.createClient(oURL.port, oURL.hostname),
 17         request = client.request('GET', oURL.pathname, { 'host': oURL.hostname });
 18 
 19     client.addListener('error', function(err) {
 20       if (err.errno === process.ECONNREFUSED) {
 21         fabric.log('ECONNREFUSED: connection refused to ' + client.host + ':' + client.port);
 22       }
 23       else {
 24         fabric.log(err.message);
 25       }
 26     });
 27 
 28     request.end();
 29     request.on('response', function (response) {
 30       var body = "";
 31       if (encoding) {
 32         response.setEncoding(encoding);
 33       }
 34       response.on('end', function () {
 35         callback(body);
 36       });
 37       response.on('data', function (chunk) {
 38         if (response.statusCode == 200) {
 39           body += chunk;
 40         }
 41       });
 42     });
 43   }

 45   fabric.util.loadImage = function(url, callback) {
 46     request(url, 'binary', function(body) {
 47       var img = new Image();
 48       img.src = new Buffer(body, 'binary');
 49       // preserving original url, which seems to be lost in node-canvas
 50       img._src = url;
 51       callback(img);
 52     });
 53   };

 55   fabric.loadSVGFromURL = function(url, callback) {
 56     url = url.replace(/^\n\s*/, '').replace(/\?.*$/, '').trim();
 57     request(url, '', function(body) {
 58       fabric.loadSVGFromString(body, callback);
 59     });
 60   };
 61 
 62   fabric.loadSVGFromString = function(string, callback) {
 63     var doc = new DOMParser().parseFromString(string);
 64     fabric.parseSVGDocument(doc.documentElement, function(results, options) {
 65       callback(results, options);
 66     });
 67   };
 68 
 69   fabric.util.getScript = function(url, callback) {
 70     request(url, '', function(body) {
 71       eval(body);
 72       callback && callback();
 73     });
 74   };

 86   /**

 93   fabric.createCanvasForNode = function(width, height) {
 94 
 95     var canvasEl = fabric.document.createElement('canvas'),
 96         nodeCanvas = new Canvas(width || 600, height || 600);
 97 
 98     // jsdom doesn't create style on canvas element, so here be temp. workaround
 99     canvasEl.style = { };
100 
101     canvasEl.width = nodeCanvas.width;
102     canvasEl.height = nodeCanvas.height;
103 
104     var canvas = fabric.Canvas || fabric.StaticCanvas;
105     var fabricCanvas = new canvas(canvasEl);
106     fabricCanvas.contextContainer = nodeCanvas.getContext('2d');
107     fabricCanvas.nodeCanvas = nodeCanvas;
108 
109     return fabricCanvas;
110   };
