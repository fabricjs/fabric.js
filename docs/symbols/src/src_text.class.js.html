282     /**
283      * Returns svg representation of an instance
284      * @method toSVG
285      * @return {string} svg representation of an instance
286      */
287     toSVG: function() {
288 
289       var textLines = this.text.split(/\r?\n/),
290           lineTopOffset = -this._fontAscent - ((this._fontAscent / 5) * this.lineHeight),
291 
292           textLeftOffset = -(this.width/2),
293           textTopOffset = (this.height/2) - (textLines.length * this.fontSize) - this._totalLineHeight,
294 
295           textAndBg = this._getSVGTextAndBg(lineTopOffset, textLeftOffset, textLines),
296           shadowSpans = this._getSVGShadows(lineTopOffset, textLines);
297 
298       return [
299         '<g transform="', this.getSvgTransform(), '">',
300           textAndBg.textBgRects.join(''),
301           '<text ',
302             (this.fontFamily ? 'font-family="\'' + this.fontFamily + '\'" ': ''),
303             (this.fontSize ? 'font-size="' + this.fontSize + '" ': ''),
304             (this.fontStyle ? 'font-style="' + this.fontStyle + '" ': ''),
305             (this.fontWeight ? 'font-weight="' + this.fontWeight + '" ': ''),
306             (this.textDecoration ? 'text-decoration="' + this.textDecoration + '" ': ''),
307             'style="', this.getSvgStyles(), '" ',
308             /* svg starts from left/bottom corner so we normalize height */
309             'transform="translate(', toFixed(textLeftOffset, 2), ' ', toFixed(textTopOffset, 2), ')">',
310             shadowSpans.join(''),
311             textAndBg.textSpans.join(''),
312           '</text>',
313         '</g>'
314       ].join('');
315     },

317     _getSVGShadows: function(lineTopOffset, textLines) {
318       var shadowSpans = [], j, i, jlen, ilen, lineTopOffsetMultiplier = 1;
319 
320       for (j = 0, jlen = this._shadows.length; j < jlen; j++) {
321         for (i = 0, ilen = textLines.length; i < ilen; i++) {
322           if (textLines[i] !== '') {
323             var lineLeftOffset = (this._boundaries && this._boundaries[i]) ? this._boundaries[i].left : 0;
324             shadowSpans.push(
325               '<tspan x="',
326               toFixed((lineLeftOffset + lineTopOffsetMultiplier) + this._shadowOffsets[j][0], 2),
327               (i === 0 ? '" y' : '" dy'), '="',
328               toFixed(lineTopOffset + (i === 0 ? this._shadowOffsets[j][1] : 0), 2),
329               '" ',
330               this._getFillAttributes(this._shadows[j].color), '>',
331               textLines[i],
332             '</tspan>');
333             lineTopOffsetMultiplier = 1;
334           } else {
335             // in some environments (e.g. IE 7 & 8) empty tspans are completely ignored, using a lineTopOffsetMultiplier
336             // prevents empty tspans
337             lineTopOffsetMultiplier++;
338           }
339         }
340       }
341       return shadowSpans;
342     },
343 
344     _getSVGTextAndBg: function(lineTopOffset, textLeftOffset, textLines) {
345       var textSpans = [ ], textBgRects = [ ], i, lineLeftOffset, len, lineTopOffsetMultiplier = 1;
346 
347       // text and background
348       for (i = 0, len = textLines.length; i < len; i++) {
349         if (textLines[i] !== '') {
350           lineLeftOffset = (this._boundaries && this._boundaries[i]) ? toFixed(this._boundaries[i].left, 2) : 0;
351           textSpans.push(
352             '<tspan x="',
353             lineLeftOffset, '" ',
354             (i === 0 ? 'y' : 'dy'), '="',
355             toFixed(lineTopOffset * lineTopOffsetMultiplier, 2) , '" ',
356             // doing this on <tspan> elements since setting opacity on containing <text> one doesn't work in Illustrator
357             this._getFillAttributes(this.fill), '>',
358             textLines[i],
359             '</tspan>'
360           );
361           lineTopOffsetMultiplier = 1;
362         } else {
363           // in some environments (e.g. IE 7 & 8) empty tspans are completely ignored, using a lineTopOffsetMultiplier
364           // prevents empty tspans
365           lineTopOffsetMultiplier++;
366         }
367 
368         if (!this.backgroundColor) continue;
369 
370         textBgRects.push(
371           '<rect ',
372             this._getFillAttributes(this.backgroundColor),
373             ' x="',
374             toFixed(textLeftOffset + this._boundaries[i].left, 2),
375             '" y="',
376             /* an offset that seems to straighten things out */
377             toFixed((lineTopOffset * i) - this.height / 2 + (this.lineHeight * 2.6), 2),
378             '" width="',
379             toFixed(this._boundaries[i].width, 2),
380             '" height="',
381             toFixed(this._boundaries[i].height, 2),
382           '"></rect>');
383       }
384       return {
385         textSpans: textSpans,
386         textBgRects: textBgRects
387       };
388     },

390     // Adobe Illustrator (at least CS5) is unable to render rgba()-based fill values
391     // we work around it by "moving" alpha channel into opacity attribute and setting fill's alpha to 1
392     _getFillAttributes: function(value) {
393       var fillColor = value ? new fabric.Color(value) : '';
394       if (!fillColor || !fillColor.getSource() || fillColor.getAlpha() === 1) {
395         return 'fill="' + value + '"';
396       }
397       return 'opacity="' + fillColor.getAlpha() + '" fill="' + fillColor.setAlpha(1).toRgb() + '"';
398     },
