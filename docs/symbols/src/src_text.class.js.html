193     /**
194      * @private
195      * @method _renderViaCufon
196      */
197     _renderViaCufon: function(ctx) {
198       var o = Cufon.textOptions || (Cufon.textOptions = { });
199 
200       // export options to be used by cufon.js
201       o.left = this.left;
202       o.top = this.top;
203       o.context = ctx;
204       o.color = this.fill;
205 
206       var el = this._initDummyElementForCufon();
207 
208       // set "cursor" to top/left corner
209       this.transform(ctx);
210 
211       // draw text
212       Cufon.replaceElement(el, {
213         engine: 'canvas',
214         separate: 'none',
215         fontFamily: this.fontFamily,
216         fontWeight: this.fontWeight,
217         textDecoration: this.textDecoration,
218         textShadow: this.textShadow,
219         textAlign: this.textAlign,
220         fontStyle: this.fontStyle,
221         lineHeight: this.lineHeight,
222         strokeStyle: this.strokeStyle,
223         strokeWidth: this.strokeWidth,
224         backgroundColor: this.backgroundColor
225       });
226 
227       // update width, height
228       this.width = o.width;
229       this.height = o.height;
230 
231       this._totalLineHeight = o.totalLineHeight;
232       this._fontAscent = o.fontAscent;
233       this._boundaries = o.boundaries;
234       this._shadowOffsets = o.shadowOffsets;
235       this._shadows = o.shadows || [ ];
236 
237       // need to set coords _after_ the width/height was retreived from Cufon
238       this.setCoords();
239     },

241     /**
242      * @private
243      * @method _render_native
244      * @param {CanvasRenderingContext2D} ctx Context to render on
245      */
246     _renderViaNative: function(ctx) {
247 
248       this.transform(ctx);
249       this._setTextStyles(ctx);
250 
251       var textLines = this.text.split(/\r?\n/);
252 
253       this.width = this._getTextWidth(ctx, textLines);
254       this.height = this._getTextHeight(ctx, textLines);
255 
256       this._renderTextBackground(ctx, textLines);
257 
258       if (this.textAlign !== 'left') {
259         ctx.save();
260         ctx.translate(this.textAlign === 'center' ? (this.width / 2) : this.width, 0);
261       }
262 
263       this._setTextShadow(ctx);
264       this._renderTextFill(ctx, textLines);
265       this.textShadow && ctx.restore();
266 
267       this._renderTextStroke(ctx, textLines);
268       if (this.textAlign !== 'left') {
269         ctx.restore();
270       }
271 
272       this._renderTextDecoration(ctx, textLines);
273       this._setBoundaries(ctx, textLines);
274       this._totalLineHeight = 0;
275 
276       this.setCoords();
277     },

279     /**

284       this._boundaries = [ ];
285

287

289         var lineLeftOffset = this._getLineLeftOffset(lineWidth);
290 
291         this._boundaries.push({

294           left: lineLeftOffset

296       }

320     /**
321      * @private
322      * @method _getTextWidth
323      */
324     _getTextWidth: function(ctx, textLines) {
325       var maxWidth = ctx.measureText(textLines[0]).width;
326 
327       for (var i = 1, len = textLines.length; i < len; i++) {
328         var currentLineWidth = ctx.measureText(textLines[i]).width;
329         if (currentLineWidth > maxWidth) {
330           maxWidth = currentLineWidth;
331         }
332       }
333       return maxWidth;
334     },

336     /**
337      * @private
338      * @method _setTextShadow
339      */
340     _setTextShadow: function(ctx) {
341       if (this.textShadow) {
342 
343         // "rgba(0,0,0,0.2) 2px 2px 10px"
344         // "rgb(0, 100, 0) 0 0 5px"
345         // "red 2px 2px 1px"
346         // "#f55 123 345 567"
347         var reOffsetsAndBlur = /\s+(-?\d+)(?:px)?\s+(-?\d+)(?:px)?\s+(\d+)(?:px)?\s*/;
348 
349         var shadowDeclaration = this.textShadow;
350         var offsetsAndBlur = reOffsetsAndBlur.exec(this.textShadow);
351         var shadowColor = shadowDeclaration.replace(reOffsetsAndBlur, '');
352 
353         ctx.save();
354         ctx.shadowColor = shadowColor;
355         ctx.shadowOffsetX = parseInt(offsetsAndBlur[1], 10);
356         ctx.shadowOffsetY = parseInt(offsetsAndBlur[2], 10);
357         ctx.shadowBlur = parseInt(offsetsAndBlur[3], 10);
358 
359         this._shadows = [{
360           blur: ctx.shadowBlur,
361           color: ctx.shadowColor,
362           offX: ctx.shadowOffsetX,
363           offY: ctx.shadowOffsetY
364         }];
365 
366         this._shadowOffsets = [[
367           parseInt(ctx.shadowOffsetX, 10), parseInt(ctx.shadowOffsetY, 10)
368         ]];
369       }
370     },

372     _renderTextFill: function(ctx, textLines) {
373       this._boundaries = [ ];
374       for (var i = 0, len = textLines.length; i < len; i++) {
375         ctx.fillText(
376           textLines[i],
377           -this.width / 2,
378           (-this.height / 2) + (i * this.fontSize * this.lineHeight)
379         );
380       }

389         for (var i = 0, len = textLines.length; i < len; i++) {
390           ctx.strokeText(

396       }

399     /**
400      * @private
401      * @_renderTextBackground
402      */
403     _renderTextBackground: function(ctx, textLines) {
404       if (this.backgroundColor) {
405         ctx.save();
406         ctx.fillStyle = this.backgroundColor;
407 
408         for (var i = 0, len = textLines.length; i < len; i++) {
409 
410           var lineWidth = ctx.measureText(textLines[i]).width;
411           var lineLeftOffset = this._getLineLeftOffset(lineWidth);
412 
413           ctx.fillRect(
414             (-this.width / 2) + lineLeftOffset,
415             (-this.height / 2) + (i * this.fontSize * this.lineHeight),
416             lineWidth,
417             this.fontSize
418           );
419         }
420         ctx.restore();
421       }
422     },

424     /**
425      * @private
426      * @method _getLineLeftOffset
427      */
428     _getLineLeftOffset: function(lineWidth) {
429       if (this.textAlign === 'center') {
430         return (this.width - lineWidth) / 2;
431       }
432       if (this.textAlign === 'right') {
433         return this.width - lineWidth;
434       }
435       return 0;
436     },
437 
438     /**
439      * @private
440      * @method _renderTextDecoration
441      */
442     _renderTextDecoration: function(ctx, textLines) {
443 
444       var halfOfVerticalBox = this._getTextHeight(ctx, textLines) / 2;
445 
446       function renderLinesAtOffset(offset) {
447         for (var i = 0, len = textLines.length; i < len; i++) {
448 
449           var lineWidth = ctx.measureText(textLines[i]).width;
450           var lineLeftOffset = this._getLineLeftOffset(lineWidth);
451 
452           ctx.fillRect(
453             (-this.width / 2) + lineLeftOffset,
454             (offset + (i * this.fontSize * this.lineHeight)) - halfOfVerticalBox,
455             lineWidth,
456             1);
457         }
458       }
459 
460       if (this.textDecoration.indexOf('underline') > -1) {
461         renderLinesAtOffset.call(this, this.fontSize);
462       }
463       if (this.textDecoration.indexOf('line-through') > -1) {
464         renderLinesAtOffset.call(this, this.fontSize / 2);
465       }
466       if (this.textDecoration.indexOf('overline') > -1) {
467         renderLinesAtOffset.call(this, 0);
468       }
469     },

484     /**

489       var el = fabric.document.createElement('pre'),
490           container = fabric.document.createElement('div');
491 
492       // Cufon doesn't play nice with textDecoration=underline if element doesn't have a parent
493       container.appendChild(el);
494 
495       if (typeof G_vmlCanvasManager == 'undefined') {
496         el.innerHTML = this.text;
497       }
498       else {
499         // IE 7 & 8 drop newlines and white space on text nodes
500         // see: http://web.student.tuwien.ac.at/~e0226430/innerHtmlQuirk.html
501         // see: http://www.w3schools.com/dom/dom_mozilla_vs_ie.asp
502         el.innerText =  this.text.replace(/\r?\n/gi, '\r');
503       }
504 
505       el.style.fontSize = this.fontSize + 'px';
506       el.style.letterSpacing = 'normal';
507 
508       return el;
509     },

549     /**
550      * Returns svg representation of an instance
551      * @method toSVG
552      * @return {string} svg representation of an instance
553      */
554     toSVG: function() {
555 
556       var textLines = this.text.split(/\r?\n/),
557           lineTopOffset = this.useNative
558             ? this.fontSize * this.lineHeight
559             : (-this._fontAscent - ((this._fontAscent / 5) * this.lineHeight)),
560 
561           textLeftOffset = -(this.width/2),
562           textTopOffset = this.useNative
563             ? this.fontSize - 1
564             : (this.height/2) - (textLines.length * this.fontSize) - this._totalLineHeight,
565 
566           textAndBg = this._getSVGTextAndBg(lineTopOffset, textLeftOffset, textLines),
567           shadowSpans = this._getSVGShadows(lineTopOffset, textLines);
568 
569       // move top offset by an ascent
570       textTopOffset += (this._fontAscent ? ((this._fontAscent / 5) * this.lineHeight) : 0);
571 
572       return [
573         '<g transform="', this.getSvgTransform(), '">',
574           textAndBg.textBgRects.join(''),
575           '<text ',
576             (this.fontFamily ? 'font-family="\'' + this.fontFamily + '\'" ': ''),
577             (this.fontSize ? 'font-size="' + this.fontSize + '" ': ''),
578             (this.fontStyle ? 'font-style="' + this.fontStyle + '" ': ''),
579             (this.fontWeight ? 'font-weight="' + this.fontWeight + '" ': ''),
580             (this.textDecoration ? 'text-decoration="' + this.textDecoration + '" ': ''),
581             'style="', this.getSvgStyles(), '" ',
582             /* svg starts from left/bottom corner so we normalize height */
583             'transform="translate(', toFixed(textLeftOffset, 2), ' ', toFixed(textTopOffset, 2), ')">',
584             shadowSpans.join(''),
585             textAndBg.textSpans.join(''),
586           '</text>',
587         '</g>'
588       ].join('');
589     },

591     _getSVGShadows: function(lineTopOffset, textLines) {
592       var shadowSpans = [], j, i, jlen, ilen, lineTopOffsetMultiplier = 1;
593 
594       if (!this._shadows || !this._boundaries) {
595         return shadowSpans;
596       }
597 
598       for (j = 0, jlen = this._shadows.length; j < jlen; j++) {
599         for (i = 0, ilen = textLines.length; i < ilen; i++) {
600           if (textLines[i] !== '') {
601             var lineLeftOffset = (this._boundaries && this._boundaries[i]) ? this._boundaries[i].left : 0;
602             shadowSpans.push(
603               '<tspan x="',
604               toFixed((lineLeftOffset + lineTopOffsetMultiplier) + this._shadowOffsets[j][0], 2),
605               ((i === 0 || this.useNative) ? '" y' : '" dy'), '="',
606               toFixed(this.useNative
607                 ? ((lineTopOffset * i) - this.height / 2 + this._shadowOffsets[j][1])
608                 : (lineTopOffset + (i === 0 ? this._shadowOffsets[j][1] : 0)), 2),
609               '" ',
610               this._getFillAttributes(this._shadows[j].color), '>',
611               fabric.util.string.escapeXml(textLines[i]),
612             '</tspan>');
613             lineTopOffsetMultiplier = 1;
614           } else {
615             // in some environments (e.g. IE 7 & 8) empty tspans are completely ignored, using a lineTopOffsetMultiplier
616             // prevents empty tspans
617             lineTopOffsetMultiplier++;
618           }
619         }
620       }
621       return shadowSpans;
622     },
623 
624     _getSVGTextAndBg: function(lineTopOffset, textLeftOffset, textLines) {
625       var textSpans = [ ], textBgRects = [ ], i, lineLeftOffset, len, lineTopOffsetMultiplier = 1;
626 
627       // text and background
628       for (i = 0, len = textLines.length; i < len; i++) {
629         if (textLines[i] !== '') {
630           lineLeftOffset = (this._boundaries && this._boundaries[i]) ? toFixed(this._boundaries[i].left, 2) : 0;
631           textSpans.push(
632             '<tspan x="',
633             lineLeftOffset, '" ',
634             (i === 0 || this.useNative ? 'y' : 'dy'), '="',
635             toFixed(this.useNative ? ((lineTopOffset * i) - this.height / 2) : (lineTopOffset * lineTopOffsetMultiplier), 2) , '" ',
636             // doing this on <tspan> elements since setting opacity on containing <text> one doesn't work in Illustrator
637             this._getFillAttributes(this.fill), '>',
638             fabric.util.string.escapeXml(textLines[i]),
639             '</tspan>'
640           );
641           lineTopOffsetMultiplier = 1;
642         } else {
643           // in some environments (e.g. IE 7 & 8) empty tspans are completely ignored, using a lineTopOffsetMultiplier
644           // prevents empty tspans
645           lineTopOffsetMultiplier++;
646         }
647 
648         if (!this.backgroundColor || !this._boundaries) continue;
649 
650         textBgRects.push(
651           '<rect ',
652             this._getFillAttributes(this.backgroundColor),
653             ' x="',
654             toFixed(textLeftOffset + this._boundaries[i].left, 2),
655             '" y="',
656             /* an offset that seems to straighten things out */
657             toFixed((lineTopOffset * i) - this.height / 2, 2),
658             '" width="',
659             toFixed(this._boundaries[i].width, 2),
660             '" height="',
661             toFixed(this._boundaries[i].height, 2),
662           '"></rect>');
663       }
664       return {
665         textSpans: textSpans,
666         textBgRects: textBgRects
667       };
668     },

670     // Adobe Illustrator (at least CS5) is unable to render rgba()-based fill values
671     // we work around it by "moving" alpha channel into opacity attribute and setting fill's alpha to 1
672     _getFillAttributes: function(value) {
673       var fillColor = value ? new fabric.Color(value) : '';
674       if (!fillColor || !fillColor.getSource() || fillColor.getAlpha() === 1) {
675         return 'fill="' + value + '"';
676       }
677       return 'opacity="' + fillColor.getAlpha() + '" fill="' + fillColor.setAlpha(1).toRgb() + '"';
678     },
