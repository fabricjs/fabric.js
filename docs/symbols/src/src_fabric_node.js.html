 14   function request(url, encoding, callback) {
 15     var oURL = URL.parse(url),
 16         client = HTTP.createClient(80, oURL.hostname),
 17         request = client.request('GET', oURL.pathname, { 'host': oURL.hostname });
 18 
 19     client.addListener('error', function(err) {
 20       if (err.errno === process.ECONNREFUSED) {
 21         fabric.log('ECONNREFUSED: connection refused to ' + client.host + ':' + client.port);
 22       }
 23       else {
 24         fabric.log(err.message);
 25       }
 26     });
 27 
 28     request.end();
 29     request.on('response', function (response) {
 30       var body = "";
 31       if (encoding) {
 32         response.setEncoding(encoding);
 33       }
 34       response.on('end', function () {
 35         callback(body);
 36       });
 37       response.on('data', function (chunk) {
 38         if (response.statusCode == 200) {
 39           body += chunk;
 40         }
 41       });
 42     });
 43   }

 45   fabric.util.loadImage = function(url, callback) {
 46     request(url, 'binary', function(body) {
 47       var img = new Image();
 48       img.src = new Buffer(body, 'binary');
 49       callback(img);
 50     });
 51   };

 53   fabric.loadSVGFromURL = function(url, callback) {
 54     url = url.replace(/^\n\s*/, '').replace(/\?.*$/, '').trim();
 55     request(url, '', function(body) {
 56       var doc = XML.parseFromString(body);
 57       fabric.parseSVGDocument(doc.documentElement, function(results, options) {
 58         callback(results, options);
 59       });
 60     });
 61   };
 62   
 63   fabric.util.getScript = function(url, callback) {
 64     request(url, '', function(body) {
 65       eval(body);
 66       callback && callback();
 67     });
 68   };

 70   fabric.Image.fromObject = function(object, callback) {

 72       var oImg = new fabric.Image(img);
 73       
 74       oImg._initConfig(object);
 75       oImg._initFilters(object);
 76       callback(oImg);
 77     });
 78   };

 80   fabric.createCanvasForNode = function(width, height) {
 81     
 82     var canvasEl = fabric.document.createElement('canvas'),
 83         nodeCanvas = new Canvas(width || 600, height || 600);
 84 
 85     // jsdom doesn't create style on canvas element, so here be temp. workaround 
 86     canvasEl.style = { };
 87 
 88     canvasEl.width = nodeCanvas.width;
 89     canvasEl.height = nodeCanvas.height;
 90 
 91     var canvas = fabric.Canvas || fabric.StaticCanvas;
 92     var fabricCanvas = new canvas(canvasEl);
 93     fabricCanvas.contextContainer = nodeCanvas.getContext('2d');
 94     fabricCanvas.nodeCanvas = nodeCanvas;
 95 
 96     return fabricCanvas;
 97   };
