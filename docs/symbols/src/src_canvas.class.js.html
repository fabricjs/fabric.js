264     /**
265      * Method that defines the actions when mouse is clic ked on canvas.
266      * The method inits the currentTransform parameters and renders all the
267      * canvas so the current image can be placed on the top canvas and the rest
268      * in on the container one.
269      * @method __onMouseDown
270      * @param e {Event} Event object fired on mousedown
271      *

273     __onMouseDown: function (e) {
274 
275       // accept only left clicks
276       var isLeftClick  = 'which' in e ? e.which == 1 : e.button == 1;
277       if (!isLeftClick && !fabric.isTouchSupported) return;
278 
279       if (this.isDrawingMode) {
280         this._prepareForDrawing(e);
281 
282         // capture coordinates immediately; this allows to draw dots (when movement never occurs)
283         this._captureDrawingPath(e);
284 
285         return;
286       }
287 
288       // ignore if some object is being transformed at this moment
289       if (this._currentTransform) return;
290 
291       var target = this.findTarget(e),
292           pointer = this.getPointer(e),
293           activeGroup = this.getActiveGroup(),
294           corner;
295 
296       if (this._shouldClearSelection(e)) {

302           left: 0
303         };
304

306       }
307       else {
308         // determine if it's a drag or rotate case
309         // rotate and scale will happen at the same time
310         this.stateful && target.saveState();

312         if (corner = target._findTargetCorner(e, this._offset)) {
313           this.onBeforeScaleRotate(target);
314         }

316         this._setupCurrentTransform(e, target);

318         var shouldHandleGroupLogic = e.shiftKey && (activeGroup || this.getActiveObject()) && this.selection;

321         }
322         else {

335     /**
336       * Method that defines the actions when mouse is hovering the canvas.
337       * The currentTransform parameter will definde whether the user is rotating/scaling/translating
338       * an image or neither of them (only hovering). A group selection is also possible and would cancel
339       * all any other type of action.
340       * In case of an image transformation only the top canvas will be rendered.
341       * @method __onMouseMove
342       * @param e {Event} Event object fired on mousemove
343       *
344       */
345     __onMouseMove: function (e) {
346 
347       if (this.isDrawingMode) {
348         if (this._isCurrentlyDrawing) {
349           this._captureDrawingPath(e);
350         }
351         return;
352       }
353 
354       var groupSelector = this._groupSelector;
355 
356       // We initially clicked in an empty area, so we draw a box for multiple selection.
357       if (groupSelector !== null) {
358         var pointer = getPointer(e);
359         groupSelector.left = pointer.x - this._offset.left - groupSelector.ex;
360         groupSelector.top = pointer.y - this._offset.top - groupSelector.ey;
361         this.renderTop();
362       }
363       else if (!this._currentTransform) {
364 
365         // alias style to elimintate unnecessary lookup
366         var style = this.upperCanvasEl.style;
367 
368         // Here we are hovering the canvas then we will determine
369         // what part of the pictures we are hovering to change the caret symbol.
370         // We won't do that while dragging or rotating in order to improve the
371         // performance.
372         var target = this.findTarget(e);
373 
374         if (!target) {
375           // image/text was hovered-out from, we remove its borders
376           for (var i = this._objects.length; i--; ) {
377             if (this._objects[i] && !this._objects[i].active) {
378               this._objects[i].setActive(false);
379             }
380           }
381           style.cursor = this.CURSOR;
382         }
383         else {
384           // set proper cursor
385           this._setCursorFromEvent(e, target);
386           if (target.isActive()) {
387             // display corners when hovering over an image
388             target.setCornersVisibility && target.setCornersVisibility(true);
389           }
390         }
391       }
392       else {
393         // object is being transformed (scaled/rotated/moved/etc.)
394         var pointer = getPointer(e),
395             x = pointer.x,
396             y = pointer.y;
397 
398         this._currentTransform.target.isMoving = true;
399 
400         if (this._currentTransform.action === 'rotate') {
401           // rotate object only if shift key is not pressed
402           // and if it is not a group we are transforming
403 
404           if (!e.shiftKey) {
405             this._rotateObject(x, y);
406 
407             this.fire('object:rotating', {
408               target: this._currentTransform.target
409             });
410           }
411           if (!this._currentTransform.target.hasRotatingPoint) {
412             this._scaleObject(x, y);
413             this.fire('object:scaling', {
414               target: this._currentTransform.target
415             });
416           }
417         }
418         else if (this._currentTransform.action === 'scale') {
419           this._scaleObject(x, y);
420           this.fire('object:scaling', {
421             target: this._currentTransform.target
422           });
423         }
424         else if (this._currentTransform.action === 'scaleX') {
425           this._scaleObject(x, y, 'x');
426 
427           this.fire('object:scaling', {
428             target: this._currentTransform.target
429           });
430         }
431         else if (this._currentTransform.action === 'scaleY') {
432           this._scaleObject(x, y, 'y');
433 
434           this.fire('object:scaling', {
435             target: this._currentTransform.target
436           });
437         }
438         else {
439           this._translateObject(x, y);
440 
441           this.fire('object:moving', {
442             target: this._currentTransform.target
443           });
444         }
445         // only commit here. when we are actually moving the pictures
446         this.renderAll();
447       }
448       this.fire('mouse:move', { target: target, e: e });

451     /**
452      * Applies one implementation of 'point inside polygon' algorithm
453      * @method containsPoint
454      * @param e { Event } event object
455      * @param target { fabric.Object } object to test against
456      * @return {Boolean} true if point contains within area of given object
457      */
458     containsPoint: function (e, target) {
459       var pointer = this.getPointer(e),
460           xy = this._normalizePointer(target, pointer),
461           x = xy.x,
462           y = xy.y;
463 
464       // http://www.geog.ubc.ca/courses/klink/gis.notes/ncgia/u32.html
465       // http://idav.ucdavis.edu/~okreylos/TAship/Spring2000/PointInPolygon.html
466 
467       // we iterate through each object. If target found, return it.
468       var iLines = target._getImageLines(target.oCoords),
469           xpoints = target._findCrossPoints(x, y, iLines);
470 
471       // if xcount is odd then we clicked inside the object
472       // For the specific case of square images xcount === 1 in all true cases
473       if ((xpoints && xpoints % 2 === 1) || target._findTargetCorner(e, this._offset)) {
474         return true;
475       }
476       return false;
477     },

481      * @method _normalizePointer

483     _normalizePointer: function (object, pointer) {
484 
485       var activeGroup = this.getActiveGroup(),
486           x = pointer.x,
487           y = pointer.y;
488 
489       var isObjectInGroup = (
490         activeGroup &&
491         object.type !== 'group' &&
492         activeGroup.contains(object)
493       );
494 
495       if (isObjectInGroup) {
496         x -= activeGroup.left;
497         y -= activeGroup.top;
498       }
499       return { x: x, y: y };

520     /**
521      * @private
522      * @method _setupCurrentTransform
523      */
524     _setupCurrentTransform: function (e, target) {
525       var action = 'drag',
526           corner,
527           pointer = getPointer(e);
528 
529       if (corner = target._findTargetCorner(e, this._offset)) {
530         action = (corner === 'ml' || corner === 'mr')
531           ? 'scaleX'
532           : (corner === 'mt' || corner === 'mb')
533             ? 'scaleY'
534             : (corner === 'mtr' || corner === 'mbr')
535               ? 'rotate'
536               : (target.hasRotatingPoint)
537                 ? 'scale'
538                 : 'rotate';
539       }
540 
541       this._currentTransform = {
542         target: target,
543         action: action,
544         scaleX: target.scaleX,
545         scaleY: target.scaleY,
546         offsetX: pointer.x - target.left,
547         offsetY: pointer.y - target.top,
548         ex: pointer.x,
549         ey: pointer.y,
550         left: target.left,
551         top: target.top,
552         theta: target.theta,
553         width: target.width * target.scaleX
554       };
555 
556       this._currentTransform.original = {
557         left: target.left,
558         top: target.top
559       };
560     },

562     _handleGroupLogic: function (e, target) {
563       if (target.isType('group')) {
564         // if it's a group, find target again, this time skipping group
565         target = this.findTarget(e, true);
566         // if even object is not found, bail out
567         if (!target || target.isType('group')) {
568           return;
569         }
570       }
571       var activeGroup = this.getActiveGroup();
572       if (activeGroup) {
573         if (activeGroup.contains(target)) {
574           activeGroup.remove(target);
575           target.setActive(false);
576           if (activeGroup.size() === 1) {
577             // remove group alltogether if after removal it only contains 1 object
578             this.discardActiveGroup();
579           }
580         }
581         else {
582           activeGroup.add(target);
583         }
584         this.fire('selection:created', { target: activeGroup, e: e });
585         activeGroup.setActive(true);
586       }
587       else {
588         // group does not exist
589         if (this._activeObject) {
590           // only if there's an active object
591           if (target !== this._activeObject) {
592             // and that object is not the actual target
593             var group = new fabric.Group([ this._activeObject,target ]);
594             this.setActiveGroup(group);
595             activeGroup = this.getActiveGroup();
596           }
597         }
598         // activate target object in any case
599         target.setActive(true);
600       }
601 
602       if (activeGroup) {
603         activeGroup.saveCoords();
604       }
605     },

645     /**
646      * @private
647      * @method _finalizeDrawingPath
648      */
649     _finalizeDrawingPath: function() {
650 
651       this.contextTop.closePath();
652 
653       this._isCurrentlyDrawing = false;
654 
655       var minX = utilMin(this._freeDrawingXPoints),
656           minY = utilMin(this._freeDrawingYPoints),
657           maxX = utilMax(this._freeDrawingXPoints),
658           maxY = utilMax(this._freeDrawingYPoints),
659           ctx = this.contextTop,
660           path = [ ],
661           xPoint,
662           yPoint,
663           xPoints = this._freeDrawingXPoints,
664           yPoints = this._freeDrawingYPoints;
665 
666       path.push('M ', xPoints[0] - minX, ' ', yPoints[0] - minY, ' ');
667 
668       for (var i = 1; xPoint = xPoints[i], yPoint = yPoints[i]; i++) {
669         path.push('L ', xPoint - minX, ' ', yPoint - minY, ' ');
670       }
671 
672       // TODO (kangax): maybe remove Path creation from here, to decouple fabric.Canvas from fabric.Path,
673       // and instead fire something like "drawing:completed" event with path string
674 
675       path = path.join('');
676 
677       if (path === "M 0 0 L 0 0 ") {
678         // do not create 0 width/height paths, as they are rendered inconsistently across browsers
679         // Firefox 4, for example, renders a dot, whereas Chrome 10 renders nothing
680         return;
681       }
682 
683       var p = new fabric.Path(path);
684 
685       p.fill = null;
686       p.stroke = this.freeDrawingColor;
687       p.strokeWidth = this.freeDrawingLineWidth;
688       this.add(p);
689       p.set("left", minX + (maxX - minX) / 2).set("top", minY + (maxY - minY) / 2).setCoords();
690       this.renderAll();
691       this.fire('path:created', { path: p });
692     },

706     /**
707      * Scales object by invoking its scaleX/scaleY methods
708      * @method _scaleObject
709      * @param x {Number} pointer's x coordinate
710      * @param y {Number} pointer's y coordinate
711      * @param by {String} Either 'x' or 'y' - specifies dimension constraint by which to scale an object.
712      *                    When not provided, an object is scaled by both dimensions equally
713      */
714     _scaleObject: function (x, y, by) {
715       var t = this._currentTransform,
716           offset = this._offset,
717           target = t.target;
718 
719       if (target.lockScalingX && target.lockScalingY) return;
720 
721       var lastLen = sqrt(pow(t.ey - t.top - offset.top, 2) + pow(t.ex - t.left - offset.left, 2)),
722           curLen = sqrt(pow(y - t.top - offset.top, 2) + pow(x - t.left - offset.left, 2));
723 
724       target._scaling = true;
725 
726       if (!by) {
727         target.lockScalingX || target.set('scaleX', t.scaleX * curLen/lastLen);
728         target.lockScalingY || target.set('scaleY', t.scaleY * curLen/lastLen);
729       }
730       else if (by === 'x' && !target.lockUniScaling) {
731         target.lockScalingX || target.set('scaleX', t.scaleX * curLen/lastLen);
732       }
733       else if (by === 'y' && !target.lockUniScaling) {
734         target.lockScalingY || target.set('scaleY', t.scaleY * curLen/lastLen);
735       }
736     },

738     /**
739      * Rotates object by invoking its rotate method
740      * @method _rotateObject
741      * @param x {Number} pointer's x coordinate
742      * @param y {Number} pointer's y coordinate
743      */
744     _rotateObject: function (x, y) {
745 
746       var t = this._currentTransform,
747           o = this._offset;
748 
749       if (t.target.lockRotation) return;
750 
751       var lastAngle = atan2(t.ey - t.top - o.top, t.ex - t.left - o.left),
752           curAngle = atan2(y - t.top - o.top, x - t.left - o.left);
753 
754       t.target.set('theta', (curAngle - lastAngle) + t.theta);
755     },

767      * @method _setCursorFromEvent
768      * @param e {Event} Event object
769      * @param target {Object} Object that the mouse is hovering, if so.
770      */
771     _setCursorFromEvent: function (e, target) {
772       var s = this.upperCanvasEl.style;
773       if (!target) {
774         s.cursor = this.CURSOR;
775         return false;
776       }
777       else {
778         var activeGroup = this.getActiveGroup();
779         // only show proper corner when group selection is not active
780         var corner = !!target._findTargetCorner
781                       && (!activeGroup || !activeGroup.contains(target))
782                       && target._findTargetCorner(e, this._offset);
783 
784         if (!corner) {
785           s.cursor = this.HOVER_CURSOR;
786         }
787         else {
788           if (corner in cursorMap) {
789             s.cursor = cursorMap[corner];
790           } else if (corner === 'mtr' || corner === 'mbr') {
791             s.cursor = this.ROTATION_CURSOR
792           } else {
793             s.cursor = this.CURSOR;
794             return false;
795           }
796         }
797       }
798       return true;
799     },

801     /**
802      * @method _drawSelection
803      * @private
804      */
805     _drawSelection: function () {
806       var groupSelector = this._groupSelector,
807           left = groupSelector.left,
808           top = groupSelector.top,
809           aleft = abs(left),
810           atop = abs(top);
811 
812       this.contextTop.fillStyle = this.selectionColor;
813 
814       this.contextTop.fillRect(
815         groupSelector.ex - ((left > 0) ? 0 : -left),
816         groupSelector.ey - ((top > 0) ? 0 : -top),
817         aleft,
818         atop
819       );
820 
821       this.contextTop.lineWidth = this.selectionLineWidth;
822       this.contextTop.strokeStyle = this.selectionBorderColor;
823 
824       this.contextTop.strokeRect(
825         groupSelector.ex + STROKE_OFFSET - ((left > 0) ? 0 : aleft),
826         groupSelector.ey + STROKE_OFFSET - ((top > 0) ? 0 : atop),
827         aleft,
828         atop
829       );
830     },

832     _findSelectedObjects: function (e) {
833       var target,
834           targetRegion,
835           group = [ ],
836           x1 = this._groupSelector.ex,
837           y1 = this._groupSelector.ey,
838           x2 = x1 + this._groupSelector.left,
839           y2 = y1 + this._groupSelector.top,
840           currentObject,
841           selectionX1Y1 = new fabric.Point(min(x1, x2), min(y1, y2)),
842           selectionX2Y2 = new fabric.Point(max(x1, x2), max(y1, y2));
843 
844       for (var i = 0, len = this._objects.length; i < len; ++i) {
845         currentObject = this._objects[i];
846 
847         if (!currentObject) continue;
848 
849         if (currentObject.intersectsWithRect(selectionX1Y1, selectionX2Y2) ||
850             currentObject.isContainedWithinRect(selectionX1Y1, selectionX2Y2)) {
851 
852           if (this.selection && currentObject.selectable) {
853             currentObject.setActive(true);
854             group.push(currentObject);
855           }
856         }
857       }
858 
859       // do not create group for 1 element only
860       if (group.length === 1) {
861         this.setActiveObject(group[0], e);
862       }
863       else if (group.length > 1) {
864         var group = new fabric.Group(group);
865         this.setActiveGroup(group);
866         group.saveCoords();
867         this.fire('selection:created', { target: group });
868       }
869 
870       this.renderAll();
871     },

873     /**
874      * Method that determines what object we are clicking on
875      * @method findTarget
876      * @param {Event} e mouse event
877      * @param {Boolean} skipGroup when true, group is skipped and only objects are traversed through
878      */
879     findTarget: function (e, skipGroup) {
880 
881       var target,
882           pointer = this.getPointer(e);
883 
884       // first check current group (if one exists)
885       var activeGroup = this.getActiveGroup();
886 
887       if (activeGroup && !skipGroup && this.containsPoint(e, activeGroup)) {
888         target = activeGroup;
889         return target;
890       }
891 
892       // then check all of the objects on canvas
893       for (var i = this._objects.length; i--; ) {
894         if (this._objects[i] && this.containsPoint(e, this._objects[i])) {
895           target = this._objects[i];
896           this.relatedTarget = target;
897           break;
898         }
899       }
900       if (target && target.selectable) {
901         return target;
902       }
903     },
