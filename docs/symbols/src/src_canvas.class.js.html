204     /**
205      * Method that defines the actions when mouse is released on canvas.
206      * The method resets the currentTransform parameters, store the image corner
207      * position in the image object and render the canvas on top.
208      * @method __onMouseUp
209      * @param {Event} e Event object fired on mouseup
210      *
211      */
212     __onMouseUp: function (e) {
213 
214       if (this.isDrawingMode && this._isCurrentlyDrawing) {
215         this._finalizeDrawingPath();
216         this.fire('mouse:up', { e: e });
217         return;
218       }
219 
220       if (this._currentTransform) {
221 
222         var transform = this._currentTransform,
223             target = transform.target;
224 
225         if (target._scaling) {
226           target._scaling = false;
227         }
228 
229         // determine the new coords everytime the image changes its position
230         var i = this._objects.length;
231         while (i--) {
232           this._objects[i].setCoords();
233         }
234

236         if (this.stateful && target.hasStateChanged()) {
237           target.isMoving = false;

240         }
241       }
242

244

246         // group selection was completed, determine its bounds
247         this._findSelectedObjects(e);
248       }

252         activeGroup.set('isMoving', false);
253         this._setCursor(this.defaultCursor);
254       }
255 
256       // clear selection
257       this._groupSelector = null;
258       this.renderAll();
259 
260       this._setCursorFromEvent(e, target);
261 
262       // fix for FF
263       this._setCursor('');
264 
265       var _this = this;
266       setTimeout(function () {
267         _this._setCursorFromEvent(e, target);

274     /**
275      * Method that defines the actions when mouse is clic ked on canvas.
276      * The method inits the currentTransform parameters and renders all the
277      * canvas so the current image can be placed on the top canvas and the rest
278      * in on the container one.
279      * @method __onMouseDown
280      * @param e {Event} Event object fired on mousedown
281      *
282      */
283     __onMouseDown: function (e) {
284 
285       // accept only left clicks
286       var isLeftClick  = 'which' in e ? e.which == 1 : e.button == 1;
287       if (!isLeftClick && !fabric.isTouchSupported) return;
288 
289       if (this.isDrawingMode) {
290         this._prepareForDrawing(e);
291 
292         // capture coordinates immediately; this allows to draw dots (when movement never occurs)
293         this._captureDrawingPath(e);
294         this.fire('mouse:down', { e: e });
295         return;
296       }
297 
298       // ignore if some object is being transformed at this moment
299       if (this._currentTransform) return;
300 
301       var target = this.findTarget(e),
302           pointer = this.getPointer(e),
303           activeGroup = this.getActiveGroup(),
304           corner;
305 
306       if (this._shouldClearSelection(e)) {
307 
308         this._groupSelector = {
309           ex: pointer.x,
310           ey: pointer.y,
311           top: 0,
312           left: 0
313         };
314 
315         this.deactivateAllWithDispatch();
316       }
317       else {
318         // determine if it's a drag or rotate case
319         // rotate and scale will happen at the same time
320         this.stateful && target.saveState();
321 
322         if (corner = target._findTargetCorner(e, this._offset)) {
323           this.onBeforeScaleRotate(target);
324         }
325 
326         this._setupCurrentTransform(e, target);
327 
328         var shouldHandleGroupLogic = e.shiftKey && (activeGroup || this.getActiveObject()) && this.selection;
329         if (shouldHandleGroupLogic) {
330           this._handleGroupLogic(e, target);
331         }
332         else {
333           if (target !== this.getActiveGroup()) {
334             this.deactivateAll();
335           }
336           this.setActiveObject(target, e);
337         }
338       }
339       // we must renderAll so that active image is placed on the top canvas
340       this.renderAll();
341 
342       this.fire('mouse:down', { target: target, e: e });
343       target && target.fire('mousedown', { e: e });

346     /**
347       * Method that defines the actions when mouse is hovering the canvas.
348       * The currentTransform parameter will definde whether the user is rotating/scaling/translating
349       * an image or neither of them (only hovering). A group selection is also possible and would cancel
350       * all any other type of action.
351       * In case of an image transformation only the top canvas will be rendered.
352       * @method __onMouseMove
353       * @param e {Event} Event object fired on mousemove
354       *
355       */
356     __onMouseMove: function (e) {
357 
358       if (this.isDrawingMode) {
359         if (this._isCurrentlyDrawing) {
360           this._captureDrawingPath(e);
361         }
362         this.fire('mouse:move', { e: e });
363         return;
364       }
365 
366       var groupSelector = this._groupSelector;
367 
368       // We initially clicked in an empty area, so we draw a box for multiple selection.
369       if (groupSelector !== null) {
370         var pointer = getPointer(e);
371         groupSelector.left = pointer.x - this._offset.left - groupSelector.ex;
372         groupSelector.top = pointer.y - this._offset.top - groupSelector.ey;
373         this.renderTop();
374       }
375       else if (!this._currentTransform) {
376 
377         // alias style to elimintate unnecessary lookup
378         var style = this.upperCanvasEl.style;
379 
380         // Here we are hovering the canvas then we will determine
381         // what part of the pictures we are hovering to change the caret symbol.
382         // We won't do that while dragging or rotating in order to improve the
383         // performance.
384         var target = this.findTarget(e);
385 
386         if (!target) {
387           // image/text was hovered-out from, we remove its borders
388           for (var i = this._objects.length; i--; ) {
389             if (this._objects[i] && !this._objects[i].active) {
390               this._objects[i].setActive(false);
391             }
392           }
393           style.cursor = this.defaultCursor;
394         }
395         else {
396           // set proper cursor
397           this._setCursorFromEvent(e, target);
398           if (target.isActive()) {
399             // display corners when hovering over an image
400             target.setCornersVisibility && target.setCornersVisibility(true);
401           }
402         }
403       }
404       else {
405         // object is being transformed (scaled/rotated/moved/etc.)
406         var pointer = getPointer(e),
407             x = pointer.x,
408             y = pointer.y;
409 
410         this._currentTransform.target.isMoving = true;
411 
412         if (this._currentTransform.action === 'rotate') {
413           // rotate object only if shift key is not pressed
414           // and if it is not a group we are transforming
415 
416           if (!e.shiftKey) {
417             this._rotateObject(x, y);
418 
419             this.fire('object:rotating', {
420               target: this._currentTransform.target
421             });
422             this._currentTransform.target.fire('rotating');
423           }
424           if (!this._currentTransform.target.hasRotatingPoint) {
425             this._scaleObject(x, y);
426             this.fire('object:scaling', {
427               target: this._currentTransform.target
428             });
429             this._currentTransform.target.fire('scaling');
430           }
431         }
432         else if (this._currentTransform.action === 'scale') {
433           this._scaleObject(x, y);
434           this.fire('object:scaling', {
435             target: this._currentTransform.target
436           });
437           this._currentTransform.target.fire('scaling');
438         }
439         else if (this._currentTransform.action === 'scaleX') {
440           this._scaleObject(x, y, 'x');
441 
442           this.fire('object:scaling', {
443             target: this._currentTransform.target
444           });
445           this._currentTransform.target.fire('scaling');
446         }
447         else if (this._currentTransform.action === 'scaleY') {
448           this._scaleObject(x, y, 'y');
449 
450           this.fire('object:scaling', {
451             target: this._currentTransform.target
452           });
453           this._currentTransform.target.fire('scaling');
454         }
455         else {
456           this._translateObject(x, y);
457 
458           this.fire('object:moving', {
459             target: this._currentTransform.target
460           });
461 
462           this._setCursor(this.moveCursor);
463 
464           this._currentTransform.target.fire('moving');
465         }
466         // only commit here. when we are actually moving the pictures
467         this.renderAll();
468       }
469       this.fire('mouse:move', { target: target, e: e });
470       target && target.fire('mousemove', { e: e });
471     },

473     /**
474      * Applies one implementation of 'point inside polygon' algorithm
475      * @method containsPoint
476      * @param e { Event } event object
477      * @param target { fabric.Object } object to test against
478      * @return {Boolean} true if point contains within area of given object
479      */
480     containsPoint: function (e, target) {
481       var pointer = this.getPointer(e),
482           xy = this._normalizePointer(target, pointer),
483           x = xy.x,
484           y = xy.y;
485 
486       // http://www.geog.ubc.ca/courses/klink/gis.notes/ncgia/u32.html
487       // http://idav.ucdavis.edu/~okreylos/TAship/Spring2000/PointInPolygon.html
488 
489       // we iterate through each object. If target found, return it.
490       var iLines = target._getImageLines(target.oCoords),
491           xpoints = target._findCrossPoints(x, y, iLines);
492 
493       // if xcount is odd then we clicked inside the object
494       // For the specific case of square images xcount === 1 in all true cases
495       if ((xpoints && xpoints % 2 === 1) || target._findTargetCorner(e, this._offset)) {
496         return true;
497       }
498       return false;
499     },

501     /**
502      * @private
503      * @method _normalizePointer
504      */
505     _normalizePointer: function (object, pointer) {
506 
507       var activeGroup = this.getActiveGroup(),
508           x = pointer.x,
509           y = pointer.y;
510 
511       var isObjectInGroup = (
512         activeGroup &&
513         object.type !== 'group' &&
514         activeGroup.contains(object)
515       );
516 
517       if (isObjectInGroup) {
518         x -= activeGroup.left;
519         y -= activeGroup.top;
520       }
521       return { x: x, y: y };
522     },

543      * @private
544      * @method _setupCurrentTransform
545      */
546     _setupCurrentTransform: function (e, target) {
547       var action = 'drag',
548           corner,
549           pointer = getPointer(e);
550 
551       if (corner = target._findTargetCorner(e, this._offset)) {
552         action = (corner === 'ml' || corner === 'mr')
553           ? 'scaleX'
554           : (corner === 'mt' || corner === 'mb')
555             ? 'scaleY'
556             : corner === 'mtr'
557               ? 'rotate'
558               : (target.hasRotatingPoint)
559                 ? 'scale'
560                 : 'rotate';
561       }
562 
563       this._currentTransform = {
564         target: target,
565         action: action,
566         scaleX: target.scaleX,
567         scaleY: target.scaleY,
568         offsetX: pointer.x - target.left,
569         offsetY: pointer.y - target.top,
570         ex: pointer.x,
571         ey: pointer.y,
572         left: target.left,
573         top: target.top,
574         theta: target.theta,
575         width: target.width * target.scaleX
576       };
577 
578       this._currentTransform.original = {
579         left: target.left,
580         top: target.top
581       };
582     },

584     _handleGroupLogic: function (e, target) {
585       if (target === this.getActiveGroup()) {
586         // if it's a group, find target again, this time skipping group
587         target = this.findTarget(e, true);
588         // if even object is not found, bail out
589         if (!target || target.isType('group')) {
590           return;
591         }
592       }
593       var activeGroup = this.getActiveGroup();
594       if (activeGroup) {
595         if (activeGroup.contains(target)) {
596           activeGroup.removeWithUpdate(target);
597           target.setActive(false);
598           if (activeGroup.size() === 1) {
599             // remove group alltogether if after removal it only contains 1 object
600             this.discardActiveGroup();
601           }
602         }
603         else {
604           activeGroup.addWithUpdate(target);
605         }
606         this.fire('selection:created', { target: activeGroup, e: e });
607         activeGroup.setActive(true);
608       }
609       else {
610         // group does not exist
611         if (this._activeObject) {
612           // only if there's an active object
613           if (target !== this._activeObject) {
614             // and that object is not the actual target
615             var group = new fabric.Group([ this._activeObject, target ]);
616             this.setActiveGroup(group);
617             activeGroup = this.getActiveGroup();
618           }
619         }
620         // activate target object in any case
621         target.setActive(true);
622       }
623 
624       if (activeGroup) {
625         activeGroup.saveCoords();
626       }
627     },

667     /**
668      * @private
669      * @method _finalizeDrawingPath
670      */
671     _finalizeDrawingPath: function() {
672 
673       this.contextTop.closePath();
674 
675       this._isCurrentlyDrawing = false;
676 
677       var minX = utilMin(this._freeDrawingXPoints),
678           minY = utilMin(this._freeDrawingYPoints),
679           maxX = utilMax(this._freeDrawingXPoints),
680           maxY = utilMax(this._freeDrawingYPoints),
681           ctx = this.contextTop,
682           path = [ ],
683           xPoint,
684           yPoint,
685           xPoints = this._freeDrawingXPoints,
686           yPoints = this._freeDrawingYPoints;
687 
688       path.push('M ', xPoints[0] - minX, ' ', yPoints[0] - minY, ' ');
689 
690       for (var i = 1; xPoint = xPoints[i], yPoint = yPoints[i]; i++) {
691         path.push('L ', xPoint - minX, ' ', yPoint - minY, ' ');
692       }
693 
694       // TODO (kangax): maybe remove Path creation from here, to decouple fabric.Canvas from fabric.Path,
695       // and instead fire something like "drawing:completed" event with path string
696 
697       path = path.join('');
698 
699       if (path === "M 0 0 L 0 0 ") {
700         // do not create 0 width/height paths, as they are rendered inconsistently across browsers
701         // Firefox 4, for example, renders a dot, whereas Chrome 10 renders nothing
702         return;
703       }
704 
705       var p = new fabric.Path(path);
706 
707       p.fill = null;
708       p.stroke = this.freeDrawingColor;
709       p.strokeWidth = this.freeDrawingLineWidth;
710       this.add(p);
711       p.set("left", minX + (maxX - minX) / 2).set("top", minY + (maxY - minY) / 2).setCoords();
712       this.renderAll();
713       this.fire('path:created', { path: p });
714     },

728     /**
729      * Scales object by invoking its scaleX/scaleY methods
730      * @method _scaleObject
731      * @param x {Number} pointer's x coordinate
732      * @param y {Number} pointer's y coordinate
733      * @param by {String} Either 'x' or 'y' - specifies dimension constraint by which to scale an object.
734      *                    When not provided, an object is scaled by both dimensions equally
735      */
736     _scaleObject: function (x, y, by) {
737       var t = this._currentTransform,
738           offset = this._offset,
739           target = t.target;
740 
741       if (target.lockScalingX && target.lockScalingY) return;
742 
743       var lastLen = sqrt(pow(t.ey - t.top - offset.top, 2) + pow(t.ex - t.left - offset.left, 2)),
744           curLen = sqrt(pow(y - t.top - offset.top, 2) + pow(x - t.left - offset.left, 2));
745 
746       target._scaling = true;
747 
748       if (!by) {
749         target.lockScalingX || target.set('scaleX', t.scaleX * curLen/lastLen);
750         target.lockScalingY || target.set('scaleY', t.scaleY * curLen/lastLen);
751       }
752       else if (by === 'x' && !target.lockUniScaling) {
753         target.lockScalingX || target.set('scaleX', t.scaleX * curLen/lastLen);
754       }
755       else if (by === 'y' && !target.lockUniScaling) {
756         target.lockScalingY || target.set('scaleY', t.scaleY * curLen/lastLen);
757       }
758     },

766     _rotateObject: function (x, y) {
767 
768       var t = this._currentTransform,
769           o = this._offset;
770 
771       if (t.target.lockRotation) return;
772 
773       var lastAngle = atan2(t.ey - t.top - o.top, t.ex - t.left - o.left),
774           curAngle = atan2(y - t.top - o.top, x - t.left - o.left);
775 
776       t.target.set('theta', (curAngle - lastAngle) + t.theta);
777     },

786     /**
787      * Sets the cursor depending on where the canvas is being hovered.
788      * Note: very buggy in Opera
789      * @method _setCursorFromEvent
790      * @param e {Event} Event object
791      * @param target {Object} Object that the mouse is hovering, if so.
792      */
793     _setCursorFromEvent: function (e, target) {
794       var s = this.upperCanvasEl.style;
795       if (!target) {
796         s.cursor = this.defaultCursor;
797         return false;
798       }
799       else {
800         var activeGroup = this.getActiveGroup();
801         // only show proper corner when group selection is not active
802         var corner = !!target._findTargetCorner
803                       && (!activeGroup || !activeGroup.contains(target))
804                       && target._findTargetCorner(e, this._offset);
805 
806         if (!corner) {
807           s.cursor = this.hoverCursor;
808         }
809         else {
810           if (corner in cursorMap) {
811             s.cursor = cursorMap[corner];
812           } else if (corner === 'mtr' && target.hasRotatingPoint) {
813             s.cursor = this.rotationCursor;
814           } else {
815             s.cursor = this.defaultCursor;
816             return false;
817           }
818         }
819       }
820       return true;
821     },

823     /**
824      * @method _drawSelection
825      * @private
826      */
827     _drawSelection: function () {
828       var groupSelector = this._groupSelector,
829           left = groupSelector.left,
830           top = groupSelector.top,
831           aleft = abs(left),
832           atop = abs(top);
833 
834       this.contextTop.fillStyle = this.selectionColor;
835 
836       this.contextTop.fillRect(
837         groupSelector.ex - ((left > 0) ? 0 : -left),
838         groupSelector.ey - ((top > 0) ? 0 : -top),
839         aleft,
840         atop
841       );
842 
843       this.contextTop.lineWidth = this.selectionLineWidth;
844       this.contextTop.strokeStyle = this.selectionBorderColor;
845 
846       this.contextTop.strokeRect(
847         groupSelector.ex + STROKE_OFFSET - ((left > 0) ? 0 : aleft),
848         groupSelector.ey + STROKE_OFFSET - ((top > 0) ? 0 : atop),
849         aleft,
850         atop

854     _findSelectedObjects: function (e) {
855       var target,
856           targetRegion,
857           group = [ ],
858           x1 = this._groupSelector.ex,
859           y1 = this._groupSelector.ey,
860           x2 = x1 + this._groupSelector.left,
861           y2 = y1 + this._groupSelector.top,
862           currentObject,
863           selectionX1Y1 = new fabric.Point(min(x1, x2), min(y1, y2)),
864           selectionX2Y2 = new fabric.Point(max(x1, x2), max(y1, y2));
865 
866       for (var i = 0, len = this._objects.length; i < len; ++i) {
867         currentObject = this._objects[i];
868 
869         if (!currentObject) continue;
870 
871         if (currentObject.intersectsWithRect(selectionX1Y1, selectionX2Y2) ||
872             currentObject.isContainedWithinRect(selectionX1Y1, selectionX2Y2)) {
873 
874           if (this.selection && currentObject.selectable) {
875             currentObject.setActive(true);
876             group.push(currentObject);
877           }
878         }
879       }
880 
881       // do not create group for 1 element only
882       if (group.length === 1) {
883         this.setActiveObject(group[0], e);
884       }
885       else if (group.length > 1) {
886         var group = new fabric.Group(group);
887         this.setActiveGroup(group);
888         group.saveCoords();
889         this.fire('selection:created', { target: group });
890       }
891 
892       this.renderAll();
893     },

895     /**
896      * Method that determines what object we are clicking on
897      * @method findTarget
898      * @param {Event} e mouse event
899      * @param {Boolean} skipGroup when true, group is skipped and only objects are traversed through
900      */
901     findTarget: function (e, skipGroup) {
902 
903       var target,
904           pointer = this.getPointer(e);
905 
906       // first check current group (if one exists)
907       var activeGroup = this.getActiveGroup();
908 
909       if (activeGroup && !skipGroup && this.containsPoint(e, activeGroup)) {
910         target = activeGroup;
911         return target;
912       }
913 
914       // then check all of the objects on canvas
915       for (var i = this._objects.length; i--; ) {
916         if (this._objects[i] && this.containsPoint(e, this._objects[i])) {
917           target = this._objects[i];
918           this.relatedTarget = target;
919           break;
920         }
921       }
922       if (target && target.selectable) {
923         return target;
924       }
925     },
