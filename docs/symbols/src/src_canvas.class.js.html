561     /**
562      * Method that defines the actions when mouse is released on canvas.
563      * The method resets the currentTransform parameters, store the image corner
564      * position in the image object and render the canvas on top.
565      * @method __onMouseUp
566      * @param {Event} e Event object fired on mouseup
567      *
568      */
569     __onMouseUp: function (e) {
570       
571       if (this.isDrawingMode && this._isCurrentlyDrawing) {
572         this._finalizeDrawingPath();
573         return;
574       }
575       
576       if (this._currentTransform) {
577         
578         var transform = this._currentTransform,
579             target = transform.target;
580             
581         if (target._scaling) {
582           target._scaling = false;
583         }
584         
585         // determine the new coords everytime the image changes its position
586         var i = this._objects.length;
587         while (i--) {
588           this._objects[i].setCoords();
589         }
590         
591         // only fire :modified event if target coordinates were changed during mousedown-mouseup
592         if (this.stateful && target.hasStateChanged()) {
593           target.isMoving = false;
594           this.fire('object:modified', { target: target });
595         }
596       }
597       
598       this._currentTransform = null;
599       
600       if (this._groupSelector) {
601         // group selection was completed, determine its bounds
602         this._findSelectedObjects(e);
603       }
604       var activeGroup = this.getActiveGroup();
605       if (activeGroup) {
606         activeGroup.setObjectsCoords();
607         activeGroup.set('isMoving', false);
608         this._setCursor(this.CURSOR);
609       }
610       
611       // clear selection
612       this._groupSelector = null;
613       this.renderAll();
614       
615       this._setCursorFromEvent(e, target);
616       
617       // fix for FF
618       this._setCursor('');
619       
620       var _this = this;
621       setTimeout(function () {
622         _this._setCursorFromEvent(e, target);
623       }, 50);
624       
625       this.fire('mouse:up', { target: target, e: e });
626     },

642     /**
643      * Method that defines the actions when mouse is clic ked on canvas.
644      * The method inits the currentTransform parameters and renders all the
645      * canvas so the current image can be placed on the top canvas and the rest
646      * in on the container one.
647      * @method __onMouseDown
648      * @param e {Event} Event object fired on mousedown
649      *
650      */
651     __onMouseDown: function (e) {
652       
653       // accept only left clicks
654       if (e.which !== 1 && !fabric.isTouchSupported) return;
655       
656       if (this.isDrawingMode) {
657         this._prepareForDrawing(e);
658         
659         // capture coordinates immediately; this allows to draw dots (when movement never occurs)
660         this._captureDrawingPath(e);
661         
662         return;
663       }
664       
665       // ignore if some object is being transformed at this moment
666       if (this._currentTransform) return;
667       
668       var target = this.findTarget(e),
669           pointer = this.getPointer(e),
670           activeGroup = this.getActiveGroup(), 
671           corner;
672       
673       if (this._shouldClearSelection(e)) {
674         
675         this._groupSelector = {
676           ex: pointer.x,
677           ey: pointer.y,
678           top: 0,
679           left: 0
680         };
681         
682         this.deactivateAllWithDispatch();
683       }
684       else {
685         // determine if it's a drag or rotate case
686         // rotate and scale will happen at the same time
687         this.stateful && target.saveState();
688         
689         if (corner = target._findTargetCorner(e, this._offset)) {
690           this.onBeforeScaleRotate(target);
691         }
692         
693         this._setupCurrentTransform(e, target);
694         
695         var shouldHandleGroupLogic = e.shiftKey && (activeGroup || this.getActiveObject());
696         if (shouldHandleGroupLogic) {
697           this._handleGroupLogic(e, target);
698         }
699         else {
700           if (target !== this.getActiveGroup()) {
701             this.deactivateAll();
702           }
703           this.setActiveObject(target);
704         }
705       }
706       // we must renderAll so that active image is placed on the top canvas
707       this.renderAll();
708       
709       this.fire('mouse:down', { target: target, e: e });
710     },

742     _setupCurrentTransform: function (e, target) {
743       var action = 'drag', 
744           corner,
745           pointer = getPointer(e);
746       
747       if (corner = target._findTargetCorner(e, this._offset)) {
748         action = (corner === 'ml' || corner === 'mr') 
749           ? 'scaleX' 
750           : (corner === 'mt' || corner === 'mb') 
751             ? 'scaleY' 
752             : 'rotate';
753       }
754       
755       this._currentTransform = {
756         target: target,
757         action: action,
758         scaleX: target.scaleX,
759         scaleY: target.scaleY,
760         offsetX: pointer.x - target.left,
761         offsetY: pointer.y - target.top,
762         ex: pointer.x,
763         ey: pointer.y,
764         left: target.left, 
765         top: target.top,
766         theta: target.theta,
767         width: target.width * target.scaleX
768       };
769       
770       this._currentTransform.original = {
771         left: target.left,
772         top: target.top

776     _handleGroupLogic: function (e, target) {
777       if (target.isType('group')) {
778         // if it's a group, find target again, this time skipping group
779         target = this.findTarget(e, true);
780         // if even object is not found, bail out
781         if (!target || target.isType('group')) {
782           return;
783         }
784       }
785       var activeGroup = this.getActiveGroup();
786       if (activeGroup) {
787         if (activeGroup.contains(target)) {
788           activeGroup.remove(target);
789           target.setActive(false);
790           if (activeGroup.size() === 1) {
791             // remove group alltogether if after removal it only contains 1 object
792             this.discardActiveGroup();
793           }
794         }
795         else {
796           activeGroup.add(target);
797         }
798         this.fire('selection:created', { target: activeGroup });
799         activeGroup.setActive(true);
800       }
801       else {
802         // group does not exist
803         if (this._activeObject) {
804           // only if there's an active object
805           if (target !== this._activeObject) {
806             // and that object is not the actual target
807             var group = new fabric.Group([ this._activeObject,target ]);
808             this.setActiveGroup(group);
809             activeGroup = this.getActiveGroup();
810           }
811         }
812         // activate target object in any case
813         target.setActive(true);
814       }
815       
816       if (activeGroup) {
817         activeGroup.saveCoords();
818       }
819     },

1031      * @method _scaleObject
1032      * @param x {Number} pointer's x coordinate
1033      * @param y {Number} pointer's y coordinate
1034      * @param by {String} Either 'x' or 'y' - specifies dimension constraint by which to scale an object. 
1035      *                    When not provided, an object is scaled by both dimensions equally
1036      */ 
1037     _scaleObject: function (x, y, by) {

1039           offset = this._offset,
1040           target = t.target;
1041       
1042       if (target.lockScalingX && target.lockScalingY) return;
1043

1045           curLen = sqrt(pow(y - t.top - offset.top, 2) + pow(x - t.left - offset.left, 2));

1057         target.lockScalingY || target.set('scaleY', t.scaleY * curLen/lastLen);

1394      * @return {Boolean} true if point contains within area of given object
1395      */

1398           xy = this._normalizePointer(target, pointer),
1399           x = xy.x, 
1400           y = xy.y;

1402       // http://www.geog.ubc.ca/courses/klink/gis.notes/ncgia/u32.html

1404

1406       var iLines = target._getImageLines(target.oCoords),
1407           xpoints = target._findCrossPoints(x, y, iLines);
1408       
1409       // if xcount is odd then we clicked inside the object
1410       // For the specific case of square images xcount === 1 in all true cases
1411       if ((xpoints && xpoints % 2 === 1) || target._findTargetCorner(e, this._offset)) {
1412         return true;
1413       }
1414       return false;
