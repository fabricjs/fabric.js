  8   /**
  9    * Removes value from an array.
 10    * Presence of value (and its position in an array) is determined via `Array.prototype.indexOf`
 11    * @static
 12    * @memberOf fabric.util
 13    * @method removeFromArray
 14    * @param {Array} array
 15    * @param {Any} value
 16    * @return {Array} original array
 17    */
 18   function removeFromArray(array, value) {
 19     var idx = array.indexOf(value);
 20     if (idx !== -1) {
 21       array.splice(idx, 1);
 22     }
 23     return array;
 24   };

 26   /**
 27    * Returns random number between 2 specified ones.
 28    * @static
 29    * @method getRandomInt
 30    * @memberOf fabric.util
 31    * @param {Number} min lower limit
 32    * @param {Number} max upper limit
 33    * @return {Number} random value (between min and max)
 34    */
 35   function getRandomInt(min, max) {
 36     return Math.floor(Math.random() * (max - min + 1)) + min;
 37   }

 77    /**
 78     * Changes value from one to another within certain period of time, invoking callbacks as value is being changed.
 79     * @method animate
 80     * @memberOf fabric.util
 81     * @param {Object} [options] Animation options
 82     * @param {Function} [options.onChange] Callback; invoked on every value change
 83     * @param {Function} [options.onComplete] Callback; invoked when value change is completed
 84     * @param {Number} [options.startValue=0] Starting value
 85     * @param {Number} [options.endValue=100] Ending value
 86     * @param {Function} [options.easing] Easing function
 87     * @param {Number} [options.duration=500] Duration of change
 88     */
 89   function animate(options) {
 90 
 91     options || (options = { });
 92 
 93     var start = +new Date(), 
 94       duration = options.duration || 500,
 95       finish = start + duration, time, pos,
 96       onChange = options.onChange || function() { },
 97       abort = options.abort || function() { return false; },
 98       easing = options.easing || function(pos) { return (-Math.cos(pos * Math.PI) / 2) + 0.5; },
 99       startValue = 'startValue' in options ? options.startValue : 0,
100       endValue = 'endValue' in options ? options.endValue : 100;
101 
102     options.onStart && options.onStart();
103 
104     var interval = setInterval(function() {
105       time = +new Date();
106       pos = time > finish ? 1 : (time - start) / duration;
107       onChange(startValue + (endValue - startValue) * easing(pos));
108       if (time > finish || abort()) {
109         clearInterval(interval);
110         options.onComplete && options.onComplete();
111      }
112     }, 10);
113 
114     return interval;
115   }

117   /**
118     * Loads image element from given url and passes it to a callback
119     * @method loadImage
120     * @memberOf fabric.util
121     * @param {String} url URL representing an image
122     * @param {Function} callback Callback; invoked with loaded image
123     * @param {Any} context optional Context to invoke callback in
124     */
125   function loadImage(url, callback, context) {
126     if (url) {
127       var img = new Image();
128       /** @ignore */
129       img.onload = function () { 
130         callback && callback.call(context, img);
131         img = img.onload = null;
132       };
133       img.src = url;
134     }
135   }
