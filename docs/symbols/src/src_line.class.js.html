1 //= require "object.class"
  2 
  3 (function() {
  4   
  5   var fabric = this.fabric || (this.fabric = { }),
  6       extend = fabric.util.object.extend;
  7       
  8   if (fabric.Line) {
  9     fabric.warn('fabric.Line is already defined');
 10     return;
 11   }
 12   
 13   /** 
 14    * @class Line
 15    * @extends fabric.Object
 16    */
 17   fabric.Line = fabric.util.createClass(fabric.Object, /** @scope fabric.Line.prototype */ {
 18     
 19     /**
 20      * @property
 21      * @type String
 22      */
 23     type: 'line',
 24     
 25     /**
 26      * Constructor
 27      * @method initialize
 28      * @param {Array} points Array of points
 29      * @param {Object} [options] Options object
 30      * @return {fabric.Line} thisArg
 31      */
 32     initialize: function(points, options) {
 33       if (!points) {
 34         points = [0, 0, 0, 0];
 35       }
 36       
 37       this.callSuper('initialize', options);
 38       
 39       this.set('x1', points[0]);
 40       this.set('y1', points[1]);
 41       this.set('x2', points[2]);
 42       this.set('y2', points[3]);
 43       
 44       this.set('width', this.x2 - this.x1);
 45       this.set('height', this.y2 - this.y1);
 46       this.set('left', this.x1 + this.width / 2);
 47       this.set('top', this.y1 + this.height / 2);
 48     },
 49     
 50     /**
 51      * @private
 52      * @method _render
 53      * @param {CanvasRenderingContext2D} ctx Context to render on
 54      */
 55     _render: function(ctx) {
 56       ctx.beginPath();
 57       
 58       // move from center (of virtual box) to its left/top corner
 59       ctx.moveTo(-this.width / 2, -this.height / 2);
 60       ctx.lineTo(this.width / 2, this.height / 2);
 61       
 62       // TODO: test this
 63       // make sure setting "fill" changes color of a line
 64       // (by copying fillStyle to strokeStyle, since line is stroked, not filled)
 65       var origStrokeStyle = ctx.strokeStyle;
 66       ctx.strokeStyle = ctx.fillStyle;
 67       ctx.stroke();
 68       ctx.strokeStyle = origStrokeStyle;
 69     },
 70     
 71     /**
 72      * Returns complexity of an instance
 73      * @method complexity
 74      * @return {Number} complexity
 75      */
 76     complexity: function() {
 77       return 1;
 78     },
 79     
 80     /**
 81      * Returns object representation of an instance
 82      * @methd toObject
 83      * @return {Object}
 84      */
 85     toObject: function() {
 86       return extend(this.callSuper('toObject'), {
 87         x1: this.get('x1'),
 88         y1: this.get('y1'),
 89         x2: this.get('x2'),
 90         y2: this.get('y2')
 91       });
 92     }
 93   });
 94   
 95   /**
 96    * List of attribute names to account for when parsing SVG element (used by `fabric.Line.fromElement`)
 97    * @static
 98    * @see http://www.w3.org/TR/SVG/shapes.html#LineElement
 99    */
100   fabric.Line.ATTRIBUTE_NAMES = 'x1 y1 x2 y2 stroke stroke-width transform'.split(' ');
101   
102   /**
103    * Returns fabric.Line instance from an SVG element
104    * @static
105    * @method fabric.Line.fromElement
106    * @param {SVGElement} element Element to parse
107    * @param {Object} [options] Options object
108    * @return {fabric.Line} instance of fabric.Line
109    */
110   fabric.Line.fromElement = function(element, options) {
111     var parsedAttributes = fabric.parseAttributes(element, fabric.Line.ATTRIBUTE_NAMES);
112     var points = [
113       parsedAttributes.x1 || 0,
114       parsedAttributes.y1 || 0,
115       parsedAttributes.x2 || 0,
116       parsedAttributes.y2 || 0
117     ];
118     return new fabric.Line(points, extend(parsedAttributes, options));
119   };
120   
121   /**
122    * Returns fabric.Line instance from an object representation
123    * @static
124    * @method fabric.Line.fromObject
125    * @param {Object} object Object to create an instance from
126    * @return {fabric.Line} instance of fabric.Line
127    */
128   fabric.Line.fromObject = function(object) {
129     var points = [object.x1, object.y1, object.x2, object.y2];
130     return new fabric.Line(points, object);
131   };
132 })();
