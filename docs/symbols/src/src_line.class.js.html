  7   var fabric = global.fabric || (global.fabric = { }),
  8       extend = fabric.util.object.extend,
  9       parentSet = fabric.Object.prototype.set,
 10       coordProps = { 'x1': 1, 'x2': 1, 'y1': 1, 'y2': 1 };
 11       
 12   if (fabric.Line) {
 13     fabric.warn('fabric.Line is already defined');
 14     return;
 15   }

 17   /** 
 18    * @class Line
 19    * @extends fabric.Object
 20    */
 21   fabric.Line = fabric.util.createClass(fabric.Object, /** @scope fabric.Line.prototype */ {
 22     
 23     /**
 24      * @property
 25      * @type String
 26      */
 27     type: 'line',
 28     
 29     /**
 30      * Constructor
 31      * @method initialize
 32      * @param {Array} points Array of points
 33      * @param {Object} [options] Options object
 34      * @return {fabric.Line} thisArg
 35      */
 36     initialize: function(points, options) {
 37       if (!points) {
 38         points = [0, 0, 0, 0];
 39       }
 40       
 41       this.callSuper('initialize', options);
 42       
 43       this.set('x1', points[0]);
 44       this.set('y1', points[1]);
 45       this.set('x2', points[2]);
 46       this.set('y2', points[3]);
 47       
 48       this._setWidthHeight();
 49     },
 50     
 51     _setWidthHeight: function() {
 52       this.set('width', (this.x2 - this.x1) || 1);
 53       this.set('height', (this.y2 - this.y1) || 1);
 54       this.set('left', this.x1 + this.width / 2);
 55       this.set('top', this.y1 + this.height / 2);
 56     },
 57     
 58     set: function(name, value) {
 59       parentSet.call(this, name, value);
 60       if (name in coordProps) {
 61         this._setWidthHeight();
 62       }
 63       return this;
 64     },
 65     
 66     /**
 67      * @private
 68      * @method _render
 69      * @param {CanvasRenderingContext2D} ctx Context to render on
 70      */
 71     _render: function(ctx) {
 72       ctx.beginPath();
 73       
 74       // move from center (of virtual box) to its left/top corner
 75       ctx.moveTo(this.width === 1 ? 0 : (-this.width / 2), this.height === 1 ? 0 : (-this.height / 2));
 76       ctx.lineTo(this.width === 1 ? 0 : (this.width / 2), this.height === 1 ? 0 : (this.height / 2));
 77       
 78       ctx.lineWidth = this.strokeWidth;
 79       
 80       // TODO: test this
 81       // make sure setting "fill" changes color of a line
 82       // (by copying fillStyle to strokeStyle, since line is stroked, not filled)
 83       var origStrokeStyle = ctx.strokeStyle;
 84       ctx.strokeStyle = ctx.fillStyle;
 85       ctx.stroke();
 86       ctx.strokeStyle = origStrokeStyle;
 87     },
 88     
 89     /**
 90      * Returns complexity of an instance
 91      * @method complexity
 92      * @return {Number} complexity
 93      */
 94     complexity: function() {
 95       return 1;
 96     },
 97     
 98     /**
 99      * Returns object representation of an instance
100      * @methd toObject
101      * @return {Object}
102      */
103     toObject: function() {
104       return extend(this.callSuper('toObject'), {
105         x1: this.get('x1'),
106         y1: this.get('y1'),
107         x2: this.get('x2'),
108         y2: this.get('y2')
109       });
110     },
111

113      * Returns svg representation of an instance

119         '<line ',
120           'x1="', this.get('x1'), '" ',
121           'y1="', this.get('y1'), '" ',
122           'x2="', this.get('x2'), '" ',

125         '/>'
126       ].join('');
127     }

130   /**
131    * List of attribute names to account for when parsing SVG element (used by `fabric.Line.fromElement`)
132    * @static
133    * @see http://www.w3.org/TR/SVG/shapes.html#LineElement
134    */
135   fabric.Line.ATTRIBUTE_NAMES = 'x1 y1 x2 y2 stroke stroke-width transform'.split(' ');
136   
137   /**
138    * Returns fabric.Line instance from an SVG element
139    * @static
140    * @method fabric.Line.fromElement
141    * @param {SVGElement} element Element to parse
142    * @param {Object} [options] Options object
143    * @return {fabric.Line} instance of fabric.Line
144    */
145   fabric.Line.fromElement = function(element, options) {
146     var parsedAttributes = fabric.parseAttributes(element, fabric.Line.ATTRIBUTE_NAMES);
147     var points = [
148       parsedAttributes.x1 || 0,
149       parsedAttributes.y1 || 0,
150       parsedAttributes.x2 || 0,
151       parsedAttributes.y2 || 0
152     ];
153     return new fabric.Line(points, extend(parsedAttributes, options));
154   };
155   
156   /**
157    * Returns fabric.Line instance from an object representation
158    * @static
159    * @method fabric.Line.fromObject
160    * @param {Object} object Object to create an instance from
161    * @return {fabric.Line} instance of fabric.Line
162    */
163   fabric.Line.fromObject = function(object) {
164     var points = [object.x1, object.y1, object.x2, object.y2];
165     return new fabric.Line(points, object);
166   };
