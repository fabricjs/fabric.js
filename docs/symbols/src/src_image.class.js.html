157     /**
158      * Returns svg representation of an instance
159      * @method toSVG
160      * @return {string} svg representation of an instance
161      */
162     toSVG: function() {
163       return '<g transform="' + this.getSvgTransform() + '">'+
164                 '<image xlink:href="' + this.getSvgSrc() + '" '+
165                   'style="' + this.getSvgStyles() + '" ' +
166                   // we're essentially moving origin of transformation from top/left corner to the center of the shape
167                   // by wrapping it in container <g> element with actual transformation, then offsetting object to the top/left
168                   // so that object's center aligns with container's left/top
169                   'transform="translate('+ (-this.width/2) + ' ' + (-this.height/2) + ')" ' +
170                   'width="' + this.width + '" ' +
171                   'height="' + this.height + '"' + '/>'+
172               '</g>';

202     /**
203      * Applies filters assigned to this image (from "filters" array)
204      * @mthod applyFilters
205      * @param {Function} callback Callback is invoked when all filters have been applied and new image is generated
206      */
207     applyFilters: function(callback) {
208 
209       if (this.filters.length === 0) {
210         this.setElement(this._originalImage);
211         callback && callback();
212         return;
213       }
214 
215       var isLikelyNode = typeof Buffer !== 'undefined' && typeof window === 'undefined',
216           imgEl = this._originalImage,
217           canvasEl = fabric.document.createElement('canvas'),
218           replacement = isLikelyNode ? new (require('canvas').Image) : fabric.document.createElement('img'),
219           _this = this;
220 
221         if (!canvasEl.getContext && typeof G_vmlCanvasManager != 'undefined') {
222           G_vmlCanvasManager.initElement(canvasEl);
223         }
224 
225       canvasEl.width = imgEl.width;
226       canvasEl.height = imgEl.height;
227 
228       canvasEl.getContext('2d').drawImage(imgEl, 0, 0, imgEl.width, imgEl.height);
229 
230       this.filters.forEach(function(filter) {
231         filter && filter.applyTo(canvasEl);
232       });
233 
234        /** @ignore */
235       replacement.onload = function() {
236         _this._element = replacement;
237         callback && callback();
238         replacement.onload = canvasEl = imgEl = null;
239       };
240       replacement.width = imgEl.width;
241       replacement.height = imgEl.height;
242 
243       if (isLikelyNode) {
244         var base64str = canvasEl.toDataURL('image/png').replace(/data:image\/png;base64,/, '');
245         replacement.src = new Buffer(base64str, 'base64');
246         _this._element = replacement;
247 
248         // onload doesn't fire in node, so we invoke callback manually
249         callback && callback();
250       }
251       else {
252         replacement.src = canvasEl.toDataURL('image/png');
253       }
254 
255       return this;
256     },

258     /**
259      * @private
260      */
261     _render: function(ctx) {
262       ctx.drawImage(
263         this.getElement(),
264         - this.width / 2,
265         -this.height / 2,

268       );

271     /**
272      * @private
273      */
274     _adjustWidthHeightToBorders: function(showBorder) {
275       if (showBorder) {
276         this.currentBorder = this.borderwidth;
277         this.width += (2 * this.currentBorder);
278         this.height += (2 * this.currentBorder);
279       }
280       else {
281         this.currentBorder = 0;
282       }
283     },

345       var sidesBorderWidth = 2 * this.currentBorder;
346 
347       this.width = 'width' in options
348         ? options.width
349         : ((this.getElement().width || 0) + sidesBorderWidth);
350 
351       this.height = 'height' in options
352         ? options.height
353         : ((this.getElement().height || 0) + sidesBorderWidth);
354     },
