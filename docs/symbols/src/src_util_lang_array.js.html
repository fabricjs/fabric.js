  6     Array.prototype.indexOf = function (searchElement /*, fromIndex */ ) {
  7       if (this === void 0 || this === null) {
  8         throw new TypeError();
  9       }
 10       var t = Object(this), len = t.length >>> 0;
 11       if (len === 0) {
 12         return -1;
 13       }
 14       var n = 0;
 15       if (arguments.length > 0) {
 16         n = Number(arguments[1]);
 17         if (n !== n) { // shortcut for verifying if it's NaN
 18           n = 0;
 19         }
 20         else if (n !== 0 && n !== (1 / 0) && n !== -(1 / 0)) {
 21           n = (n > 0 || -1) * Math.floor(Math.abs(n));
 22         }
 23       }
 24       if (n >= len) {
 25         return -1;
 26       }
 27       var k = n >= 0 ? n : Math.max(len - Math.abs(n), 0);
 28       for (; k < len; k++) {
 29         if (k in t && t[k] === searchElement) {
 30           return k;
 31         }
 32       }
 33       return -1;
 34     }

149   function max(array, byProperty) {
150     if (!array || array.length === 0) return undefined;
151     
152     var i = array.length - 1, 
153         result = byProperty ? array[i][byProperty] : array[i];

159       }
160     }
