1 /**
  2  * @namespace
  3  */
  4 fabric.Observable = {
  5   
  6   /**
  7    * Observes specified event
  8    * @method observe
  9    * @param {String} eventName
 10    * @param {Function} handler
 11    */
 12   observe: function(eventName, handler) {
 13     if (!this.__eventListeners) {
 14       this.__eventListeners = { };
 15     }
 16     // one object with key/value pairs was passed
 17     if (arguments.length === 1) {
 18       for (var prop in eventName) {
 19         this.observe(prop, eventName[prop]);
 20       }
 21     }
 22     else {
 23       if (!this.__eventListeners[eventName]) {
 24         this.__eventListeners[eventName] = [ ];
 25       }
 26       this.__eventListeners[eventName].push(handler);
 27     }
 28   },
 29   
 30   /**
 31    * Stops event observing for a particular event handler
 32    * @method stopObserving
 33    * @param {String} eventName
 34    * @param {Function} handler
 35    */
 36   stopObserving: function(eventName, handler) {
 37     if (!this.__eventListeners) {
 38       this.__eventListeners = { };
 39     }
 40     if (this.__eventListeners[eventName]) {
 41       fabric.util.removeFromArray(this.__eventListeners[eventName], handler);
 42     }
 43   },
 44   
 45   /**
 46    * Fires event with an optional memo object
 47    * @method fire
 48    * @param {String} eventName
 49    * @param {Object} [memo]
 50    */
 51   fire: function(eventName, memo) {
 52     if (!this.__eventListeners) {
 53       this.__eventListeners = { }
 54     }
 55     var listenersForEvent = this.__eventListeners[eventName];
 56     if (!listenersForEvent) return;
 57     for (var i = 0, len = listenersForEvent.length; i < len; i++) {
 58       // avoiding try/catch for perf. reasons
 59       listenersForEvent[i]({ memo: memo });
 60     }
 61   }
 62 };
